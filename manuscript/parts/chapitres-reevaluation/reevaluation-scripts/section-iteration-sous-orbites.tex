
\section{Gestion des iterations sur les sous-orbites}
\label{sec:extension-scripts-iteration}

\begin{itemize}
  \item présentation des boucles de sous-orbites
  \item extension de la spécification paramétrique avec une boîte d'itération qui contient
  \begin{itemize}
    \item la boîte du script avec son nom et le nom persistant d'une orbite
    \item une boîte \textit{for} avec le nom persistant de la sous-orbite d'itération
    \item les enregistrements de l'opération itérée avec les noms persistant de chaque sous-orbite
  \end{itemize}
  \item extension des DAG avec l'ajout
  \begin{itemize}
    \item ajout d'une branche pour remonter l'histoire du paramètre d'entrée du script
    \item ajout d'une branche pour remonter l'histoire de la sous-orbite d'itération
  \end{itemize}
\end{itemize}
