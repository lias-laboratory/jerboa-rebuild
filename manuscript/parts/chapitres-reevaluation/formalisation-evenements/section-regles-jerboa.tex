\begin{proposition}
	Soit une règle $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de
	filtrage $\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte $G$, et un type
	d'orbite $\orbit{\omega}$.

	Soit $v$ un nœud de $R$. L'orbite $\underline{R}\orbit{\omega}(v)$ est créé dans la
	règle si et seulement si toutes les orbites $G\orbit{\omega}(m((v,v')))$ sont créés
	dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous les nœuds
	$v'$ de l'orbite d'instanciation $G\orbit{l(h)}(\underline{m}(h))$.
\end{proposition}

\begin{proof}
	Soit $v$ un nœud de $R$ tel que l'orbite $\underline{R}\orbit{\omega}(v)$ est créée
	par la règle, c'est-à-dire tel que
	$\underline{R}\orbit{\omega}(v) \cap \underline{L} = \emptyset$. Soit
	$O = G\orbit{l(h)}(\underline{m}(h))$, l'orbite d'instanciation.

	Soit $w$ un nœud quelconque de l'orbite $\underline{R}\orbit{\omega}(v)$. $w$ est un
	nœud créé de la règle de $R \setminus L$. Donc, par construction de la règle instanciée,
	tous les nœuds $(w,v')$ pour tous les nœuds $v'$ de l'orbite d'instanciation
	$O$, sont des nœuds créés dans la règle instanciée $r(\underline{m})$.

	Donc toutes les orbites $R(O)\orbit{\omega}((v,v'))$ pour tous les nœuds $v'$ de
	l'orbite d'instanciation $O$ sont des orbites créées dans la règle instanciée
	$r(\underline{m})$. Et donc, d'après la proposition \ref{prop:base-rule-create},
	toutes les orbites $G\orbit{\omega}(m((v,v')))$ sont créés dans la transformation
	directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous les nœuds $v'$ de l'orbite
	d'instanciation $O$.
	\\

	Réciproquement, supposons que toutes les orbites $H\orbit{\omega}(m'((v,v')))$ sont créés dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous les nœuds $v'$ de l'orbite d'instanciation $O$.
	Donc, d'après la proposition \ref{prop:base-rule-create}, toutes les orbites $R(O)\orbit{\omega}((v,v'))$ pour tous les nœuds $v'$ de l'orbite d'instanciation $O$ sont des orbites créées dans la règle instanciée $r(\underline{m})$.
	Donc par construction de l'instanciation, $\underline{R}\orbit{\omega}(v)$ est une orbite créée dans la règle $r$.
\end{proof}

\subsection{Non-changement}

Dans l'insertion de sommet figure \textbf{fig:rule-insert-vertex}, le nœud $n0$
(en noir) filtre partiellement les orbites sommet aux extrémités de l'arête sur
laquelle est inséré le nouveau sommet. Ces orbites sont inchangées par
l'insertion du nouveau sommet.

\begin{proposition} Soit une règle $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$,
	un morphisme de filtrage $\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte
	$G$, et un type d'orbite $\orbit{\omega}$.

	Soit $v$ un nœud de $L$. L'orbite $\underline{L}\orbit{\omega}(v)$ est inchangée dans
	la règle si et seulement si toutes les orbites $G\orbit{\omega}(m((v,v')))$ sont
	inchangées dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$, pour tous
	les nœuds $v'$ de l'orbite d'instanciation $G\orbit{l(h)}(\underline{m}(h))$.
\end{proposition}

\begin{proof} Soit $v$ un nœud préservé de $L \cap R$ tel que l'orbite
	$\underline{L}\orbit{\omega}(v)$ est inchangée dans la règle $r~: L \longrightarrow R$. Soit
	$O = G\orbit{l(h)}(\underline{m}(h))$, l'orbite d'instanciation.

	Soit $w$ un nœud quelconque de l'orbite $\underline{L}\orbit{\omega}(v)$. $w$ est un
	nœud inchangé de la règle de $L \cap R$. Donc, par construction de la règle
	instanciée, tous les nœuds $(w,v')$ pour tous les nœuds $v'$ de l'orbite
	d'instanciation $O$, sont des nœuds inchangés dans la règle instanciée
	$r(\underline{m})$. Or une orbite $G\orbit{\omega}(m((v,v')))$ n'est pas
	nécessairement entièrement filtrée, c'est-à-dire
	$L(O)\orbit{\omega}((v,v')) \subset G\orbit{\omega}(m((v,v')))$ et
	$R(O)\orbit{\omega}((v,v')) \subset H\orbit{\omega}(m'((v,v')))$. Par définition, la partie non
	filtrée est aussi inchangée
	$G\orbit{\omega}(m((v,v'))) \setminus L(O)\orbit{\omega}((v,v')) = H\orbit{\omega}(m'((v,v'))) \setminus R(O)\orbit{\omega}((v,v'))$.
	Donc toutes les orbites $L(O)\orbit{\omega}((v,v'))$ pour tous les nœuds $v'$ de
	l'orbite d'instanciation $O$ sont des orbites inchangées dans la règle
	instanciée $r(\underline{m})$. Et donc, d'après la proposition
	\ref{prop:base-rule-unchange}, toutes les orbites $G\orbit{\omega}(m((v,v')))$ sont
	inchangées dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$, pour tous
	les nœuds $v'$ de l'orbite d'instanciation $O$. \\

	Réciproquement, supposons que toutes les orbites $G\orbit{\omega}(m((v,v')))$ sont
	inchangées dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous
	les nœuds $v'$ de l'orbite d'instanciation $O$. Donc d'après la proposition
	\ref{prop:base-rule-unchange}, toutes les orbites $L(O)\orbit{\omega}((v,v'))$ pour
	tous les nœuds $v'$ de l'orbite d'instanciation $O$ sont des orbites inchangées
	dans la règle instanciée $r(\underline{m})$. Donc par construction de
	l'instanciation, $\underline{L}\orbit{\omega}(v)$ est une orbite inchangée dans la
	règle $r$.
\end{proof}

\begin{definition}
	Soit une règle $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de filtrage
	$\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte $G$, et un type d'orbite

	Soit $b$ un brin de $G$ non filtré. L'orbite $G\orbit{\omega}(b)$ n'est \emph{pas affectée} par l'application de $r$ si $G\orbit{\omega}(b) \cap \underline{m}(L) = \emptyset$ et \emph{inchangée} sinon.

\end{definition}

\begin{proposition}
	Soit une règle $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de
	filtrage $\underline{m}~: \underline{L} \rightarrow R$ sur une G-carte $G$, et un type d'orbite
	$\orbit{\omega}$.

	Soit un nœud préservé $v_1$ de $L$ et $R$, $\underline{L}\orbit{\omega}(v_1)$ est scindé dans la règle si :
	\begin{itemize}
		\item la scission est due à des arcs explicites, c'est-à-dire si $\underline{L}\orbit{\omega}(v_1)$ est scindée dans $\underline{R}$~;

		\item la scission est due à des arcs implicites, c'est-à-dire si il existe un nœud $v$ de $\underline{L}\orbit{\omega}(v_1)$ dont l'étiquette $l(v)$ a pour k-ème position une dimension de $\orbit{\omega}$, et que pour tous les nœuds $v'$ de $\underline{R}\orbit{\omega}(v_1)$ l'étiquette de $l(v')$ a pour k-ème position est un "$\_$" ou une dimension qui n'appartient pas à $\orbit{\omega}$.
	\end{itemize}

	Soit deux nœuds préservés $v_1$ et $v_2$ de $L$ et $R$. Notons $O = G\orbit{l(h)}(\underline{m}(h))$ l'orbite d'instanciation.
	\begin{enumerate}

		\item
		      Si $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$ dans la règle du à des arcs explicites.
		      Et si l'orbite $\underline{R}\orbit{\omega}(v_1)$ est entièrement filtrée, c'est-à-dire que pour tout nœud $v$ de $\underline{R}\orbit{\omega}(v_1)$ et pour toute dimension $i$ de $\orbit{\omega}$, $v$ est la source d'un $i$-arc implicite ou explicite dans~$R$.
		      Alors l'orbite $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2, v')))$ dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud $v'$ de l'orbite d'instanciation $O$.

		\item Si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites $\underline{R}\orbit{\omega}(v_1)$ et
		      $\underline{R}\orbit{\omega}(v_2)$ dans la règle due à un arc explicite.
		      Si les orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$
		      sont toutes deux partiellement filtrées.
		      Alors, pour chaque nœud $v'$ de l'orbite d'instanciation $O$, soit $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites
		      $H\orbit{\omega}(m((v_1, v')))$ et $H\orbit{\omega}(m((v_2,v')))$, soit l'orbite est modifiée, dans la transformation directe
		      $G \xlongrightarrow{r, \underline{m}} H$.

		\item Si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est scindée dans la règle du au k-ième arc implicite.
		      Alors pour tout nœud $v'_1$ de l'orbite d'instanciation $O$ et $v'_2$ le $i$-voisin de $v'_1$ dans $O$ avec $i$ la dimension à la k-ième position de $l(h)$, $G\orbit{\omega}(m((v_1,v'_1)))$ est soit scindée en deux orbites
		      $H\orbit{\omega}(m((v_1, v'_1)))$ et $H\orbit{\omega}(m((v_1,v'_2)))$, soit modifiée en une orbite unique $H\orbit{\omega}(m((v_1, v'_1))) = H\orbit{\omega}(m((v_1,v'_2)))$ dans la transformation directe
		      $G \xlongrightarrow{r, \underline{m}} H$.
	\end{enumerate}

\end{proposition}

\begin{proof}
	Soit deux nœuds préservés $v_1$ et $v_2$ de $L$ et $R$.
	\\

	Cas 1 :

	Supposons que $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$ dans la règle du à des arcs explicites.
	Alors, par définition, il existe un chemin $e_1 e_2 ... e_j$ dans $L$ de source $v_1$, de cible $v_2$ et étiqueté par $i_1 i_2 ... i_j$ sur $\orbit{\omega}$, mais il n'existe pas de tel chemin $f_1 f_2 ... f_k$ dans $R$ de source $v_1$, de cible $v_2$ et étiqueté sur $\orbit{\omega}$.

	Donc dans la règle instanciée, pour tout nœud $v'$ de l'orbite d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$, il existe un chemin $(e_1,v') (e_2,v') ... (e_j,v')$ dans le motif gauche de la règle instanciée $L(O)$ dont la source est $(s(e_1),v') = (v_1, v')$, la cible est $(t(e_j), v') = (v_2, v')$ et étiqueté par le même mot $i_1 i_2 ... i_j$ sur $\orbit{\omega}$.

	Supposons que pour un nœud $v'$ de l'orbite d'instanciation $O$, il existe un chemin $(f_1,v') (f_2,v') ... (f_k,v')$ dans le motif droit de la règle instanciée $R(O)$ dont la source est $(s(f_1),v') = (v_1, v')$, la cible est $(t(e_j), v') = (v_2, v')$ et étiqueté sur $\orbit{\omega}$.
	Alors, par construction de l'instanciation, le chemin $f_1 f_2 ... f_k$ dans $R$ de source $v_1$, de cible $v_2$ et étiqueté sur $\orbit{\omega}$ existe dans la règle de départ. Ce qui est contradictoire avec l'hypothèse. Donc il n'existe pas de chemin $(f_1,v') (f_2,v') ... (f_k,v')$ dans le motif droit de la règle instanciée ayant les propriétés attendues.
	Donc $R(O)\orbit{\omega}((v_1,v'))$ et $R(O)\orbit{\omega}((v_2,v'))$ sont deux $\orbit{\omega}$-orbites distinctes.

	Donc d'après la proposition \ref{prop:base-rule_split}, si $\underline{R}\orbit{\omega}(v_1)$ et donc $R(O)\orbit{\omega}((v_1, v'))$ est entièrement filtrée, alors l'orbite $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2, v')))$ dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud $v'$ de l'orbite d'instanciation $O$.
	\\

	Cas 2 :

	Supposons que $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$ dans la règle du à des arcs explicites.

	Pour les mêmes raisons que dans le cas 1, $R(O)\orbit{\omega}((v_1,v'))$ et $R(O)\orbit{\omega}((v_2,v'))$ sont deux $\orbit{\omega}$-orbites distinctes dans la règles instanciée, pour tout nœud $v'$ de l'orbite d'instanciation $O$.

	Donc d'après la proposition \ref{prop:base-rule_split}, si $R\orbit{\omega}(v_1)$ et donc $R(O)\orbit{\omega}((v_1, v'))$ est partiellement filtrée, alors l'orbite $G\orbit{\omega}(m((v_1,v')))$ est soit scindée en deux orbites $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2, v')))$, soit modifiée, dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud $v'$ de l'orbite d'instanciation $O$.
	\\

	Cas 3 :

	Supposons que $\underline{L}\orbit{\omega}(v_1)$ est scindée due au k-ième arc implicite. Soit $v'_1$ un nœud de l'orbite d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$ et $v'_2$ le $i$-voisin de $v'_1$ dans $O$, avec $i$ la k-ième dimension de l'étiquette $l(h)$ du nœud d'accroche $h$ de $L$.

	Par hypothèse, l'un des nœuds $v$ de $\underline{L}\orbit{\omega}(v_1)$ a une k-ième dimension $j$ dans l'orbite $\orbit{\omega}$ ($j \in \orbit{\omega}$), alors que tous des noeuds de $\underline{R}\orbit{\omega}(v_1)$ ont leur k-ième dimension hors de $\orbit{\omega}$.
	Donc les nœuds $(v, v'_1)$ et $(v, v'_2)$ sont $j$-voisins dans l'instanciation $L(O)$, donc $(v, v'_1)$ et $(v, v'_2)$ sont dans la même $\orbit{\omega}$-orbite dans $L(O)$, et donc $(v_1, v'_1)$ et $(v_2, v'_2)$ aussi.

	Nous avons plusieurs cas :
	\begin{itemize}
		\item si $(v_1, v'_1)$ et $(v_2, v'_2)$ sont dans deux $\orbit{\omega}$-orbites différentes de $R(O)$, d'après la proposition \ref{prop:base-rule_split},
		      selon les cas, $H\orbit{\omega}(m'((v_1, v'_1)))$ et $H\orbit{\omega}(m'((v_1,v'_2)))$ sont dans deux orbites différentes ou non de $H$, avec $H$ le résultats de la transformation directe $G \xlongrightarrow{r, \underline{m}} H$ ;
		      autrement dit $G\orbit{\omega}(m((v_1, v'_1))$ est scindée en $H\orbit{\omega}(m'((v_1, v'_1)))$ et $H\orbit{\omega}(m'((v_1,v'_2)))$ ou juste modifiée par la transformation directe $G \xlongrightarrow{r, \underline{m}} H$ ;

		\item si au contraire $(v_1, v'_1)$ et $(v_2, v'_2)$ sont dans la même $\orbit{\omega}$-orbites de $R(O)$ (car il existe un chemin $e_1 e_2 ... e_l$ de $O$ de source $v'_1$ et de cible $v'_2$ qui ne comprend pas l'étiquette $j$, ce chemin est réécrit en un chemin $(w, e_1) (w,e_2) ... (w, e_l)$ de $R(O)$ dont toutes les étiquettes sont dans $\orbit{\omega}$),
		      d'après la proposition \ref{prop:base-rule-modify}, alors $G\orbit{\omega}(m((v_1, v'_1))$ est modifiée par la transformation directe $G \xlongrightarrow{r, \underline{m}} H$.
	\end{itemize}

\begin{proposition}
	Soit une règle $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de filtrage $\underline{m}~: \underline{L} \rightarrow R$ sur une G-carte $G$, et un type d'orbite $\orbit{\omega}$.

	Soit deux nœuds préservés $v_1$ et $v_2$ de $L$ et $R$, $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées dans la règle~:

	\begin{itemize}
		\item si la fusion est due à des arcs explicites, c'est-à-dire si $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées dans $\underline{R}$~;

		\item si la fusion est due à des arcs implicites, c'est-à-dire si pour tout nœud de $\underline{L}\orbit{\omega}(v_1)$, l'étiquette $l(v)$ a pour $k$-ème position un "\_" ou une
		      dimension qui n'appartient pas à $\orbit{\omega}$, et qu'il existe un nœud $v'$ de $\underline{R}\orbit{\omega}(v_1)$ dont l'étiquette $l(v')$ a pour $k$-ème position une dimension
		      de $\orbit{\omega}$.
	\end{itemize}

	Soit deux nœuds préservés $v_1$ et $v_2$ de $L$ et $R$. Notons $O = G\orbit{l(h)}(\underline{m}(h))$ l'orbite d'instanciation.

	\begin{enumerate}
		\item Si $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées en une orbite $\underline{R}\orbit{\omega}(v_1)$ dans la règle du à des arcs explicites.
		      Et si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est entièrement filtrée, c'est-à-dire que pour tout nœud $v$ de $\underline{L}\orbit{\omega}(v_1)$ et pour toute dimension $i$ de $\orbit{\omega}$, $v$ est la source d'un $i$-arc implicite
		      ou explicite dans $L$. Alors les orbites $G\orbit{\omega}(m((v_1,v')))$ et $G\orbit{\omega}(m((v_2,v')))$ sont fusionnées en une orbite $H\orbit{\omega}(m'((v_1,v')))$ dans la transformation directe
		      $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud $v'$ de l'orbite d'instanciation $O$.

		\item Si les orbites $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées en une orbite $\underline{R}\orbit{\omega}(v_1)$ dans la règle du à des arcs explicites.
		      Si les orbites $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont toutes deux partiellement filtrées. Alors, pour chaque nœud $v'$ de l'orbite
		      d'instanciation $O$, soit $G\orbit{\omega}(m((v_1,v')))$ et $G\orbit{\omega}(m((v_2,v')))$ sont fusionnées en une orbite
		      $H\orbit{\omega}(m'((v_1,v')))$, soit elles sont deux parties d'une même orbite modifiée dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$.

		\item Si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est fusionnées dans la règle du au $k$-ème arc implicite.
		      Alors pour tout nœud $v_1'$ de l'orbite d'instanciation $O$ et $v_2'$ le $i$-voisin de $v_1'$ dans $O$ avec $i$ la dimension à la $k$-ème position
		      de $l(h)$, $G\orbit{\omega}(m((v_1,v_1')))$ et $G\orbit{\omega}(m((v_1,v_2')))$ sont soit fusionnées en une orbite $H\orbit{\omega}(m'((v_1,v_1')))$, soit modifiées en une orbite unique
		      $H\orbit{\omega}(m'((v_1,v_1'))) = H\orbit{\omega}(m'((v_1,v_2')))$ dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$
	\end{enumerate}
\end{proposition}

\begin{proof}
	Soit deux nœuds préservés $v_1$ et $v_2$ de $L$ et $R$
	\begin{enumerate}
		\item Supposons que $\underline{L}\orbit{\omega}(v_1)$ et
		      $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées en une orbite
		      $\underline{R}\orbit{\omega}(v_1)$ dans la règle du à des arcs
		      explicites. Alors, par définition, il n'existe pas de chemin
		      $e_1 e_2 \dots e_j$ dans $L$ de source $v_1$, de cible $v_2$ et
		      étiqueté par $i_1 i_2 \dots i_j$ sur $\orbit{\omega}$, mais il existe un
		      chemin $f_1 f_2 \dots f_k$ dans $R$ de source $v_1$, de cible $v_2$
		      et étiqueté sur $\orbit{\omega}$.

		      Donc dans la règle instanciée, pour tout nœud $v'$ de l'orbite d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$, il n'existe pas de chemin $(e_1, v')(e_2, v')\dots(e_j,v')$ dans le
		      motif gauche de la règle instanciée $L(O)$ dont la source est $(s(e_1),v') = (v_1,v')$, la cible est $(t(e_j),v') = (v_2,v')$ et étiqueté par le même mot $i_1 i_2 \dots i_j$ sur $\orbit{\omega}$.

		      Supposons que pour tout nœud $v'$ de l'orbite d'instanciation $O$, il n'existe pas de chemin $(f_1,v')(f_2,v')\dots(f_k,v')$ dans le motif droit de la règle instanciée $R(O)$ dont la source
		      est $(s(f_1),v') = (v_1,v')$, la cible est $(t(f_k),v') = (v_2, v')$ et étiqueté sur $\orbit{\omega}$.
		      Alors, par construction de l'instanciation, le chemin $f_1 f_2 \dots f_k$ dans $R$ de source $v_1$,
		      de cible $v_2$ et étiqueté sur $\orbit{\omega}$ n'existe pas dans la règle de départ.
		      Ce qui est contradictoire avec l'hypothèse. Donc il existe un chemin $(f_1,v')(f_2,v')\dots(f_k,v')$ dans le
		      motif droit de la règle instanciée ayant les propriétés attendues.
		      Donc $R(O)\orbit{\omega}((v_1,v'))$ et $R(O)\orbit{\omega}((v_2,v'))$ sont bien une seule et même $\orbit{\omega}$-orbite.

		      Donc d'après la proposition \ref{prop:base-rule-merge}, si $\underline{L}\orbit{\omega}(v_1)$ et donc $L(O)\orbit{\omega}((v_1,v'))$ est entièrement filtrée, alors les orbites $G\orbit{\omega}(m((v_1,v')))$ et
		      $G\orbit{\omega}(m((v_2,v')))$ sont fusionnées en une orbite $H\orbit{\omega}(m'((v_1,v')))$ dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud $v'$ de l'orbite
		      d'instanciation $O$.

		\item Supposons que $\underline{L}\orbit{\omega}(v_1)$ et
		      $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées en une orbite
		      $\underline{R}\orbit{\omega}(v_1)$ dans la règle du à des arcs
		      explicites.

		      Pour les mêmes raisons que dans le cas $1$, $L(O)\orbit{\omega}((v_1,v'))$ et $L(O)\orbit{\omega}((v_2,v'))$ sont deux $\orbit{\omega}$-orbites distinctes dans la règle instanciée, pour tout nœud $v'$ de l'orbite d'instanciation $O$.

		      Donc d'après la proposition \ref{prop:base-rule-merge}, si $\underline{L}\orbit{\omega}(v_1)$ et donc $L(O)\orbit{\omega}((v_1,v'))$ est partiellement filtrée, alors les orbites
		      $G\orbit{\omega}(m((v_1,v')))$ et $G\orbit{\omega}(m((v_2,v')))$ sont soit fusionnées en une orbite $H\orbit{\omega}(m'((v_1,v')))$, soit modifiées, dans la transformation directe
		      $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud $v'$ de l'orbite d'instanciation $O$.

		\item Supposons que $\underline{L}\orbit{\omega}(v_1)$ est fusionnée due au
		      $k$-ème arc implicite. Soit $v_1'$ un nœud de l'orbite
		      d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$ et
		      $v_2'$ le $i$-voisin de $v_1'$ dans $O$, avec $i$ la $k$-ème
		      dimension de l'étiquette $l(h)$ du nœud d'accroche $h$ de $L$.

		      Par hypothèse, tous les noeuds de $\underline{L}\orbit{\omega}(v_1)$ ont une k-ième dimension soit supprimée, soit renommée hors de $\orbit{\omega}$.
		      Au contraire, il existe un nœud $v$ de $\underline{R}\orbit{\omega}(v_1)$ dont la k-ième dimension est renommée en $j$ dans $\orbit{\omega}$. Donc les nœuds $(v, v'_1)$ et $(v, v'_2)$ sont $j$-voisins dans l'instanciation $R(O)$, donc $(v, v'_1)$ et $(v, v'_2)$ sont dans la même $\orbit{\omega}$-orbite dans $R(O)$, et donc $(v_1, v'_1)$ et $(v_2, v'_2)$ aussi.

		      Nous avons plusieurs cas :
		      \begin{itemize}
			      \item si $(v_1, v'_1)$ et $(v_2, v'_2)$ sont deux $\orbit{\omega}$-orbites différentes de $L(O)$, d'après la proposition \ref{prop:base-rule_split},
			            selon les cas, $G\orbit{\omega}(m((v_1, v'_1)))$ et $G\orbit{\omega}(m((v_1,v'_2)))$ sont dans deux orbites différentes ou non de $G$ ;
			            autrement dit $G\orbit{\omega}(m((v_1, v'_1))$ et $G\orbit{\omega}(m((v_2, v'_2))$ sont fusionnées en une orbite $H\orbit{\omega}(m'((v_1, v'_1)))$ ou l'unique orbite $G\orbit{\omega}(m((v_1, v'_1)) = G\orbit{\omega}(m((v_2, v'_2))$ est juste modifiée par la transformation directe $G \xlongrightarrow{r, \underline{m}} H$ ;

			      \item si au contraire $(v_1, v'_1)$ et $(v_2, v'_2)$ sont dans la même $\orbit{\omega}$-orbites de $L(O)$ (car il existe un chemin $e_1 e_2 ... e_l$ de $O$ de source $v'_1$ et de cible $v'_2$ qui ne comprend pas l'étiquette $j$, ce chemin est réécrit un chemin $(w, e_1) (w,e_2) ... (w, e_l)$ de $L(O)$ dont toutes les étiquettes sont dans $\orbit{\omega}$),
			            d'après la proposition \ref{prop:base-rule-modify}, alors $G\orbit{\omega}(m((v_1, v'_1))$ est modifiée par la transformation directe $G \xlongrightarrow{r, \underline{m}} H$.
		      \end{itemize}
	\end{enumerate}
\end{proof}

\begin{definition}

	Une \emph{orbite d'origine} de $\underline{R}\orbit{\omega}(n)$ dans $L$ est une orbite $\underline{L}\orbit{o'}(n')$ où :
	\begin{itemize}
		\item où $n'$ est un nœud de $L$,

		\item $\orbit{o'}$ est l'ensemble des dimensions  $i$, où $i$ est renommé dans un arc implicite de $n'$ ; c'est-à-dire que $(l_L(n') \mapsto l_R(n))(i) = j \in \orbit{\omega}$

\end{definition}
