\subsection[Rappels sur les graphes]{Rappels sur les graphes}
\label{sec:etat-art-rappels-graphes}


Les structures topologiques peuvent être représentées par des graphes dont les
arcs établissent des relations d'appartenance, entre plusieurs nœuds, à une même
structure topologique. %
Ces structures peuvent être, par exemple, des cellules telles que les volumes,
faces, arêtes et sommets lorsque le modèle de représentation est en dimension
3. %
De tels graphes sont appelés des graphes topologiques. %

\begin{definition}[Graphes topologiques]
	\label{def:graphe-topologique}
	Soit $n \in \mathbb{N}$ une dimension. Un \emph{graphe topologique} de dimension $n$ $G=(V,E,s,t,\alpha)$ est défini par:
	\begin{itemize}
		\item un ensemble $V$ de nœuds;
		\item un ensemble $E$ d'arcs;
		\item deux fonctions source $s : E \to V$ et cible $t : E \to V$ qui lient les
		      arcs aux nœuds et une fonction d'étiquetage $\alpha : E \to [0,n]$. Soit
		      l'arc $e \in E$~; $s(e)$, $t(e)$ sont appelés respectivement le nœud
		      source et le nœud cible de $e$ et $\alpha(e)$ est son étiquette.
	\end{itemize}
\end{definition}

Les principales notions ensemblistes peuvent être étendues sur les graphes.

\begin{definition}[Inclusion, union, intersection, différence]
	\label{def:graphe-notions-ensembles}
	Soient $G = (V,E,s,t,\alpha)$ et $G'=(V',E',s',t',\alpha')$ deux graphes tels que
	$s_{|E \cap E'} = s'_{|E \cap E'}$, $t_{|E \cap E'} = t'_{|E \cap E'}$ et
	$\alpha_{|E \cap E'} = \alpha'_{|E \cap E'}$.

	\begin{itemize}
		\item $G$ est \textbf{\emph{inclus}} dans $G'$, noté $G \subset G'$, si $V \subset V'$,
		      $E \subset E'$, $s'_{|E} = s$, $t'_{|E} = t$ et $\alpha'_{|E} = \alpha$.

		\item L'\textbf{\emph{intersection}} $G \cap G'$ est le graphe
		      $(V \cap V', E \cap E', s_{|E \cap E'}, t_{|E \cap E'}, \alpha_{|E \cap E'})$.

		\item L'\textbf{\emph{union}} $G \cup G'$ est le graphe
		      $(V \cup V', E \cup E', s^{\prime\prime}, t^{\prime\prime}, \alpha^{\prime\prime})$, où $s^{\prime\prime}_{|E} = s$,
		      $s^{\prime\prime}_{|E'} = s'$, $t^{\prime\prime}_{|E} = t$, $t^{\prime\prime}_{|E'} = t'$, $\alpha^{\prime\prime}_{|E} = \alpha$,
		      $\alpha^{\prime\prime}_{|E'} = \alpha'$.

		\item La \textbf{\emph{différence}} $G \setminus G'$ est la structure
		      $(V \setminus V', E \setminus E', s_{|E \setminus E'}, t_{|E \setminus E'}, \alpha_{|E \setminus E'})$.

		\item Soit $\emptyset$ le \textbf{\emph{graphe vide}}, c'est-à-dire l'unique graphe
		      ayant un ensemble vide de nœuds et d'arcs.
	\end{itemize}
\end{definition}

Ces propriétés permettent, à leur tour, de définir la notion de sous-graphe.

\begin{definition}[Sous-graphe]
	\label{def:sous-graphe}
	Soient $G=(V,E,s,t,\alpha)$ et $G'=(V',E',s',t',\alpha')$ deux graphes.%

	\noindent$G'$ est un \textbf{\emph{sous-graphe}} de $G$ si et seulement si, $G \subset G'$.
\end{definition}

De même, nous pouvons définir la notion de chemin dans les graphes. %
Intuitivement, un chemin est un ensemble fini des arêtes traversées lors d'un
parcours depuis un nœud $v_{i}$ jusqu'à un nœud $v_{k}$. %

\begin{definition}[Chemin]
	\label{def:graphes-chemin}
	Un \textbf{\emph{chemin}} $e_{1}e_{2} \dots e_{k}$ dans un graphe $G$ est
	une suite d'arcs tels que pour $1 \leq i < k$, $t(e_{i}) = s(e_{i+1})$.
	La source d'un tel chemin est $s(e_{1})$, sa cible est $t(e_{k})$ et son
	étiquette est le mot $\alpha(e_{1}) \alpha(e_{2}) \dots \alpha(e_{k})$.


	Si $s(e_{1}) = t(e_{k})$, le chemin est appelé \emph{cycle}.
\end{definition}

\subsection[Les cartes généralisées]{Les cartes généralisées}
\label{sec:etat-art-model-regles-gcartes}


Les \emph{cartes généralisées} (ou «~G-cartes~»)
\cite{lienhardt1991topological,Damiand-Lienhardt14} permettent la représentation
de variétés géométriques (avec ou sans bord) basées sur des structures
topologiques cellulaires de dimension $n$.

\begin{definition}[G-Carte]
	\label{def:gcarte}
	Une \textbf{\emph{G-carte}} de dimension $n \in \mathbb{N}$ est un graphe topologique
	$G = (V,E,s,t,\alpha)$ de dimension $n$ qui vérifie les contraintes suivantes~:
	\begin{itemize}
		\item \textbf{\emph{non-orientation}}~: pour tout arc
		      $e \in E$, il existe un arc inverse $e' \in E$ tel que
		      $s(e') = t(e)$, $t(e') = s(e)$ et $\alpha(e') = \alpha(e)$~;
		\item \textbf{\emph{arcs incidents}}~: pour toute dimension $i \in [0,n]$ et pour tout nœud $v \in V$,
		      il existe exactement un arc $e \in E$ d'étiquette $\alpha(e)=i$ et de source
		      $s(e)=v$~;
		      % \xavier{C'est ici qu'on peut écrire que les G-cartes sont étiquetées par des dimensions topologiques.}
		\item \textbf{\emph{cycle}}~: pour toutes dimensions $i, j \in [0,n]^{2}$ telles que $i+2 \leq j$, et
		      pour tout nœud $v \in V$, il existe un cycle, étiqueté par \emph{$ijij$}, de source
		      $v$.
	\end{itemize}
\end{definition}

\begin{definition}[Orbite]
	Soient $G$ une G-carte de dimension $n$, où $n \in \mathbb{N}$, $o$ un sous-ensemble de $[0, n]$ et $b$ un brin de $G$.

	L'\emph{orbite} $G\orbit{o}(b)$ incidente à $b$ et de type $o$,
	ou $\orbit{o}$-orbite, est le sous-graphe de $G$ contenant $b$, les brins
	de $G$ atteignables à partir de $b$ par des liaisons étiquetés sur $o$,
	ainsi que ces liaisons. %
	Lorsqu'il n'y a pas d'ambiguïté sur le graphe $G$
	considéré, on notera une telle orbite $b\orbit{o}$.
\end{definition}

\subsubsection[Concepts]{Concepts}
\label{sec:etat-art-model-regles-regles-transformation-concepts}

Deux concepts sont nécessaires pour définir les règles de transformation de
graphe ainsi que leurs applications~: les morphismes et unions disjointes de
graphes. %

~

Le morphisme $m$ entre deux graphes $G$ et $G'$, est l'application
définie respectivement sur les nœuds et les arcs de $G$ vers les nœuds et les
arcs de $G'$, qui préserve les sources, cibles et étiquettes des arcs. %


\begin{definition}[Morphisme]
	Soient $G = (V,E,s,t,\alpha)$
	et $G' = (V',E',s',$ $t',\alpha')$ deux graphes.
	%et $m : G \to G'$ un morphisme.
	Un \emph{morphisme}  $m: G \to G'$  est un couple de fonctions $m_{V} : V \to V'$,
	$m_{E} : E \to E'$ préservant les sources, cibles et étiquettes de G
	dans G', tel que pour tout $e \in E$, $m_{V}(s(e)) = s'(m_{E}(e))$,
	$m_{V}(t(e)) = t'(m_{E}(e))$ et $\alpha(e) = \alpha'(m_{E}(e))$.

	\noindent Un morphisme $G \to G'$ est un~:
	\begin{itemize}
		\item morphisme d'\emph{inclusion}, noté $\iota : G \hookrightarrow G'$ ou simplement $G \hookrightarrow G'$, si $\iota_{V}$ et $\iota_{E}$ sont les applications identités.
		\item morphisme \emph{injectif} si ses fonctions $m_{V}$ et $m_{E}$ sont injectives.
		\item \emph{isomorphisme} si ses fonctions $m_{V}$ et $m_{E}$ sont des bijections.
	\end{itemize}
\end{definition}

% Intuitivement, l'union disjointe de deux graphes est l'union simple de ceux-ci
% lorsque leur intersection est nulle. %
%
\begin{definition}[Union disjointe]
	Soient $G = (V,E,s,t,\alpha)$ et $G'=(V',E',s',t',\alpha')$ deux graphes.

	L'\emph{union disjointe} $G \sqcup G'$ est définie à isomorphisme près comme suit.
	Soient $m: G \to H$ et $m': G' \to H'$ deux isomorphismes tels que $H$ et $H'$ soient disjoints, c'est-à-dire $H \cap H' = \emptyset$.
	L'union disjointe de  $G$ et $G'$ est l'union simple de $H$ et $H'$, plus formellement $G \sqcup G' = H \cup H'$.
\end{definition}

\subsubsection[Règles instanciées]{Règles instanciées}
\label{sec:etat-art-model-regles-regles-inst}

\begin{figure}[ht]
	\centering
	\includegraphics[width=.6\textwidth]{application-regle-v2}
	\caption{Application d'une règle de triangulation}
	\label{fig:etat-art-model-regles-inst-application}
\end{figure}

Une règle de transformation de graphe se compose de deux membres. %
Un membre gauche pour filtrer tout ou partie d'un graphe et un membre droit pour
transformer l'élément filtré par le membre gauche. %
% C'est ce qu'illustre la règle dans le cadre vert de la
% Figure~\ref{fig:etat-art-model-regles-inst-application} dont le membre gauche
% est une face carrée et dont le membre droit est la même face carrée qui a été
% triangulée. %

Plus formellement, une règle $r$ est définie par un couple de graphes $L$ et $R$
tel que $L \cap R \hookrightarrow L$ et $L \cap R \hookrightarrow R$. %
C'est-à-dire que pour toute arête $e \in L \cap R$, $e$ a les mêmes sources, cibles
et étiquettes dans $L$ et $R$. %
Une manière simple d'assurer cette condition est que $L$ et $R$ n'aient pas
d'arête commune. %
$L$ est le motif \emph{filtré} et $R$ est le motif \emph{transformé}. %
La syntaxe d'une règle est introduite par \cite{poudr2009} et est notée
$r: L \longrightarrow R$. %

L'application d'une règle $r: L \longrightarrow R$ sur un graphe $G$ le long d'un morphisme
de filtrage $m: L \rightarrow G$ produit un graphe transformé $H$, où $m'$ est un
morphisme injectif qui étend $m$ sur $R$. %
%
Intuitivement, $m(L \setminus R)$ est supprimé dans $G$, puis $m'(R \setminus L)$ est ajouté
pour produire la transformation directe $H$. %
%
$m'$ et $H$ sont définis à isomorphisme près, c'est-à-dire au nom près des
nouveaux brins. %
L'injectivité de $m'$ garantie que les nouveaux brins de $H$ sont bien des brins
distincts les uns des autres et distincts de ceux de $G$. %
L'application d'une règle peut être définie formellement dans la théorie des
catégories~\cite{poudr2009,pascual2022topological}. %
% \maxime{thèse de Thomas Bellet aussi ?}

La Figure~\ref{fig:etat-art-model-regles-inst-application} présente la règle de
triangulation d'une face quadrangulaire et son application. %
Dans cet exemple, le morphisme $m : L \to G$ (respectivement $m' : R \to H$) est
l'identité. %

\begin{definition}[Application d'une règle de transformation de graphe]
	Soient $r: L \longrightarrow R$ une règle et $m : L \rightarrow G$ un morphisme de filtrage
	injectif. %

	L'\emph{application} $G \xlongrightarrow{(r, m)} H$ de $r$ sur $G$ le long
	de $m$ est définie par $H = (G \setminus m(L))\sqcup m'(R)$, où $m'$ est la restriction
	de $m$ sur $L \setminus R$, puis son extension sur $R$ telle que $m'$ soit
	injective. %
	$m'$, et donc $H$ est unique à isomorphisme près. %
	$H$ est la transformation directe de $G$ par la règle $r$ le long du
	morphisme de filtrage $m$. %
\end{definition}

Des conditions syntaxiques sur les règles définies dans
\cite{poudr2009,pascual2022topological} permettent de garantir la préservation
des G-cartes. %
Jerboa et son éditeur vérifient automatiquement que
les règles éditées respectent ces conditions syntaxiques. %

\begin{theoreme}[Préservation de la cohérence topologique]
	\label{theo:topo-coherence-base-rule}
	\begin{itemize}
		\item \emph{Condition de non-orientation}~: $L$ et $R$ sont
		      \emph{non-orientés}.
		\item \emph{Conditions d'arcs incidents}~: pour toute dimension
		      $i \in [0,n]$,
		      \begin{itemize}
			      \item un nœud préservé de $L \cap R$ est la source d'un arc
			            étiqueté $i$ dans $L$ si et seulement si il est la source
			            d'un arc étiqueté $i$ dans $R$~;
			      \item tout nœud supprimé de $L \setminus R$ et tout nœud créé de $R \setminus L$
			            est la source d'un unique arc étiqueté $i$.
		      \end{itemize}
		\item \emph{Conditions de cycles}~: pour tout couple $(i,j)$ tel que
		      $0 \leq i \leq i + 2 \leq j \leq n$,
		      \begin{itemize}
			      \item un nœud créé de $R \setminus L$ est la source d'un cycle étiqueté
			            $ijij$;
			      \item si un nœud préservé de $L \cap R$ est source d'un cycle
			            étiqueté $ijij$ dans $R$, il est source d'un cycle
			            étiqueté $ijij$ dans $L$~;
			      \item si un nœud préservé de $L \cap R$ n'est pas la source d'un
			            cycle étiqueté $ijij$ dans $L$, alors les arcs
			            d'étiquettes $i$ et $j$, dont il est la source, sont
			            préservés dans $R$.
		      \end{itemize}
	\end{itemize}
\end{theoreme}

\begin{remarque}
	Dans la suite, nous utiliserons uniquement des règles qui vérifient ces
	conditions de préservation des G-cartes. %
	Ainsi, lorsque $G$ est une G-carte, sa transformation directe $H$ l'est
	aussi. %
\end{remarque}

En pratique, les morphismes de filtrage ne sont pas donnés \textit{in extenso},
mais à partir \highlight{\st{du mapping} $\to$ de l'accrochage} d'un sous-ensemble
de brins de $L$ appelés brins d'accroche (\emph{hooks} en anglais). %
Par exemple, dans la Figure~\ref{fig:etat-art-model-regles-inst-application}, le
nœud $h$ de $L$ est son brin d'accroche. % noté par un double cercle. %
Son association au brin $h$ de $G$ défini un unique morphisme de filtrage $m$
grâce à la condition d'arcs incidents. %
Pour cela, les règles comportent un brin d'accroche par composante connexe de
$L$.


\subsubsection[Règles Jerboa]{Règles Jerboa}
\label{sec:etat-art-model-regles-regles-jerboa}


Comme nous l'avons vu, les motifs des règles instanciées sont des parties des
objets transformés. %
Ainsi, les règles instanciées déterminent complètement la topologie des parties
modifiées. %
Le langage de règles Jerboa \cite{belhaouari2014jerboa}, avec la paramétrisation
des motifs par des orbites, permet de concevoir des règles plus générales
permettant, par exemple, de filtrer puis transformer des faces indépendamment de
leur nombre d'arêtes. %

\begin{definition}[Motif avec orbite]
	Un \emph{motif avec orbite} est un graphe $M=(V,E,s,t,\alpha,l)$, où $(V,E,s,t,\alpha)$
	est un graphe topologique classique, $l : V \rightarrow ([0,n] \cup \{ \_ \})^*$ est une
	fonction d'étiquetage des nœuds par des mots sur les dimensions topologiques
	et le symbole "$\_$", telle que toutes les étiquettes des nœuds du motif
	soient des mots de la même longueur.
\end{definition}

Notons $\underline{M} = (V, E, s, t, \alpha)$ le graphe topologique correspondant au
motif $M$ privé de ses étiquettes de nœuds. %
Prenons le mot $013$, que l'on note $\orbit{013}$ et qui désigne l'orbite face,
un exemple est donné dans la
Figure~\ref{fig:etat-art-model-regle-regles-jerboa-motif-gauche-triangulation}. %
De même, le mot $0\_3$ est noté $\orbit{0\_3}$ et désigne l'orbite face
précédente dont tous les $1$-arcs sont supprimés, un exemple est donné avec le
nœud $n_{0}$ dans la
Figure~\ref{fig:etat-art-model-regle-regles-jerboa-motif-droit-triangulation}. %
Ces étiquettes désignent la modification des orbites par renommage et
suppression des arcs. %
Une étiquette construite sur $[0,n]$, c'est-à-dire sans suppression, est appelée
\emph{orbite pleine}. %
Une \emph{fonction de renommage} $\orbit{o} \mapsto \orbit{o'}$, d'une orbite pleine
$\orbit{o_1 o_2 ... o_k}$, vers une orbite $\orbit{o'_1 o'_2 ... o'_k}$ définit
le renommage des étiquettes topologiques $o_1 \mapsto o'_1, o_2 \mapsto o'_2$ \ldots
$o_k \mapsto o'_k$. %

\begin{definition}[Renommage d'une orbite]
	Le \emph{renommage} $(\orbit{o} \mapsto \orbit{o'})(G\orbit{o}(v))$ d'une $\orbit{o}$-orbite d'un graphe topologique $G=(V,E,s,t,\alpha)$ est le graphe $(V,E',s',t',\alpha')$ défini par :
	\begin{itemize}
		\item $E'$ est le sous-ensemble des arêtes de $E$ qui n'ont pas été
		      supprimées, c'est-à-dire
		      $E'=\{e \in E \mid (\orbit{o} \mapsto \orbit{o'})(\alpha(e)) \neq \_ \}$,

		\item $s'$ et $t'$ sont les restrictions respectives de $s$ et $t$ sur
		      $E'$,

		\item $\alpha'$ est définie sur $E'$ pour toute arête $e \in E'$ par
		      $\alpha'(e) = (\orbit{o} \mapsto \orbit{o'})(\alpha(e))$.
	\end{itemize}
\end{definition}

L'instanciation d'un motif $M=(V,E,s,t,\alpha,l)$ dont toutes les étiquettes d'orbites ont la même longueur que $\orbit{o}$, par une orbite $G\orbit{o}(v)=(V',E',s',t',\alpha')$ d'un graphe topologique $G$ est définie par le graphe topologique $M(G\orbit{o}(v))=(V'',E'',s'',t'',\alpha'')$~:
\begin{itemize}
	\item $V^{\prime\prime}= V \times V'$,

	\item $E''=E''_i \cup E''_e$ comprend les \emph{arcs implicites} issus de l'instanciation des nœuds du motif $E''_i = \{ (v, e') \in V \times E' \mid (\orbit{o} \mapsto \orbit{\alpha(v)})(\alpha'(e')) \neq \_ \}$ et les \emph{arcs explicites} issus du motif $E''_e = \{ (e, v') \in E \times V' \}$,

	\item tout arc implicite $(v,e') \in E''_i$ a pour source $s''((v,e'))=(v,s'(e'))$, pour cible $t''((v,e'))=(v,t'(e'))$ et pour étiquette $\alpha''((v,e'))=(\orbit{o} \mapsto \alpha(v))(\alpha'(e'))$,

	      et tout arc explicite $(e,v') \in E''_e$, a pour source $s''((e,v')) = ((s(e),v')$, pour cible $t''((e,v')) = ((t(e),v')$, et pour étiquette $\alpha''((e, v')) = \alpha(e)$.
\end{itemize}

\begin{definition}[Règle transformation de graphe Jerboa]
	Une \emph{règle} Jerboa  $r: L \longrightarrow R$  est définie par un couple  de motifs avec
	orbites $L$ et $R$ dont toutes  les étiquettes d'orbites ont la même longueur,
	tels    que   $\underline{L}    \cap    \underline{R}    \hookrightarrow   \underline{L}$    et
	$\underline{L} \cap \underline{R} \hookrightarrow \underline{R}$,  et dont un sous-ensemble des
	nœuds  de $L$  sont  appelés  nœuds d'accroche  (\emph{hooks}  en anglais)  et
	portent des étiquettes d'orbites pleines, c'est-à-dire sans "$\_$".
\end{definition}

L'application d'une règle Jerboa est l'application de son instanciation, qui est
une règle instanciée. %
Cela nécessite tout d'abord de rechercher dans l'objet à transformer l'orbite
d'instanciation de la règle, d'instancier la règle, de construire le morphisme
de filtrage entre le motif gauche instancié et l'objet, puis finalement
d'appliquer la règle instanciée sur l'objet pour obtenir la transformation
directe. %

\begin{definition}[Orbite d'instanciation]
	Soit $\underline{m} : \underline{L} \rightarrow G$ un morphisme de filtrage sur une
	G-carte $G$, c'est-à-dire un morphisme injectif tel que les nœuds d'accroche
	désignent des orbites isomorphes à renommage près. %
	Plus précisément, pour un nœud d'accroche $h$ de $L$,
	$O = G\orbit{l(h)}(\underline{m}(h))$ est l'orbite d'instanciation. %
	Et si la règle $r$ possède un autre nœud d'accroche $h'$, alors
	$O' = G\orbit{l(h')}(\underline{m}(h'))$ est isomorphe à renommage près,
	c'est-à-dire que $(l(h) \mapsto l(h'))(O)$ est isomorphe à $O'$.
\end{definition}

\begin{definition}[Filtrage de l'orbite d'instanciation]
	Le morphisme de filtrage $m : L(O) \rightarrow G$, s'il existe, est le morphisme qui
	vérifie les propriétés suivantes, pour $L = (V, E, s, t, \alpha, l)$ et
	$O = (V', E', s', t', \alpha')$~:
	\begin{itemize}
		\item pour tout nœud $v \in V$ de $L$,
		      $m_V((v,\underline{m}(h))) = \underline{m}(v)$~;

		\item pour tout arc implicite $(v,e')$, avec $v \in V$ un nœud de $L$ et
		      $e' \in E'$ un arc de l'orbite $O$ non supprimé (c'est-à-dire tel que
		      $\orbit{l(h)} \mapsto \orbit{l(v)} (\alpha'(e')) \neq \_$)~: $m_E((v,e'))$ est
		      l'unique arc de $G$ de source $m_V((v, s'(e')))$ et d'étiquette
		      $\orbit{l(h)} \mapsto \orbit{l(v)}(\alpha'(e'))$ ; et
		      $m_V(t((v, e'))) = t'(m_E((v,e')))$~;

		\item pour tout arc explicite $(e, v')$, avec $e \in E$ un arc de $L$ et
		      $v' \in V'$ un nœud de l'orbite $O$~: $m_E((e, v'))$ est l'unique arc de
		      source $m_V((s(e), v'))$ et d'étiquette $\alpha(e)$~; et
		      $m_V(t(e,v')) = t'(m_E(e,v'))$.
	\end{itemize}
\end{definition}

\begin{remarque}
	Si le morphisme de filtrage $m$ existe, il est unique en raison de la
	contrainte d'arc incident de la G-carte $G$ (définition \ref{def:gcarte}) et
	de la condition d'arc incident du motif $L$.
\end{remarque}

Comme pour les règles de base, des conditions syntaxiques suivantes sur les règles permettent de garantir la préservation de contraintes des G-cartes lors des transformations directes \cite{belle2012}.

\begin{theoreme}[Préservation de la cohérence topologique~: Règles Jerboa]
	Soit une règle $r~: L \longrightarrow R$, $G$ une G-carte et $\underline{m} : \underline{L} \rightarrow G$.

	La transformation directe $G \xlongrightarrow{r, \underline{m}} H$ produit une G-carte si les conditions de préservation de la cohérence suivantes sont vérifiées :
	\begin{itemize}
		\item \emph{condition de non orientation} : $L$ et $R$ sont non orientés ;

		\item pour toute dimension $i$, un nœud $v$ est source d'un arc implicite si l'étiquette d'orbite de $v$ comprend $i$~;

		      \emph{condition d'arcs incidents} : pour toute dimension $i \in [0, n]$, la condition d'arcs incidents est celle des règles de base en considérant simultanément les arcs implicites et explicites~:
		      \begin{itemize}
			      \item un nœud préservé de $L \cap R$ est  source d'un $i$-arc dans $L$ (implicite ou explicite) si et seulement s'il est  source d'un $i$-arc dans $R$ (implicite ou explicite)~;
			      \item tout nœud supprimé de $L \setminus R$ et tout nœud ajouté de $R \setminus L$ est la source d'exactement un $i$-arc (implicite ou explicite) ;
		      \end{itemize}

		\item un nœud $v$ est la source d'un $ijij$-cycle avec $i+2 \geq j$ si :
		      \begin{itemize}
			      \item $v$ est un nœud de $L$ qui contient $i$ et $j$ dans son étiquette d'orbite ;
			      \item $v$ est un nœud de $R$ qui contient $i$ et $j$ dans son étiquette d'orbite et que ces dernières renomment un cycle de $L$, c'est-à-dire tel qu'il existe un nœud $v'$ de $L$ avec $l(v) \mapsto l(v') (i) +2 \geq l(v) \mapsto l(v') (j)$ ou $l(v) \mapsto l(v') (j) +2 \geq l(v) \mapsto l(v') (i)$ ;
			      \item $v$ est la source d'un $i$-arc de cible $v'$ (dans $L$ ou $R$) tel que les deux nœuds $v$ et $v'$ ont $j$ dans leurs étiquettes d'orbite non renommé, c'est à dire tel que $l(v) \mapsto l(v') (j) = j$ ;
			      \item inversement, $v$ est la source d'un $j$-arc de cible $v'$ tel que les deux nœuds $v$ et $v'$ ont $i$ dans leurs étiquettes d'orbites non renommés ;
		      \end{itemize}

		      \emph{condition de cycle} : pour toutes dimensions $i$ et $j$ tel que $i+2 \geq j$, la condition de cycle est celle des règles de base en considérant les cycles ayant des arcs implicites ou explicites :
		      \begin{itemize}
			      \item un nœud créé $v$ de $R \setminus L$ est la source d'un $ijij$-cycle ;
			      \item si un nœud préservé de $L \cap R$ est la source d'un $ijij$-cycle dans $L$, alors il est la source d'un $ijij$-cycle dans $R$ ;
			      \item si un nœud préservé de $L \cap R$ n'est pas la source d'un $ijij$-cycle dans $L$, alors les $i$-arcs et $j$-arcs dont il est la source dans $L$ sont préservés dans $R$.
		      \end{itemize}
	\end{itemize}
\end{theoreme}

	Les règles instanciées sont aussi des règles Jerboa sur l'orbite vide $\orbit{}$.
	\highlight{ou : les règles Jerboa sur l'orbite vide $\orbit{}$ expriment explicitement des règles instanciées}
\end{remarque}
