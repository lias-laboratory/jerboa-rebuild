\begin{frame}
	\frametitle{Limitations des approches existantes}

	\begin{itemize}[<+->]
		\item édition limitée aux paramètres géométriques
		\item manque d'homogénéité dans la nomination
		\item suivi partiel des cellules
		\item conception généralement limitée à la 3D
	\end{itemize}
\end{frame}
