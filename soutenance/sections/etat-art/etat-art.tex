\begin{frame}
	\frametitle{Limitations des approches existantes}

	\begin{itemize}
		\item édition limitée aux paramètres géométriques,
		\item manque d'homogénéité dans la nomination,
		\item suivi partiel des cellules,
		\item conception généralement limitée à la 3D
		      % \item (à compléter en fonction des contributions)
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Objectifs}
	Conception et développement d'un système de réévaluation permettant le rejeu
	d'un processus constructif grâce à~: % (autant d'objectifs que de limitations)
	\begin{itemize}
		\item l'édition étendue (ajout, suppression et réorganisation des
		      opérations)~;
		\item un schéma de nomination homogène~;
		      % \item de limiter  les coûts (calcul et mémoire) d'analyses et les risques
		      %       d'erreurs d'appariement~;
	\end{itemize}
	notamment pour la construction d'objets génériques. %
\end{frame}
