%% Plus simple que lstlisting, sans verbatim et sans mathescape (je voulais uniformiser mon affichage de PN\subscript{xyz}=[x'y'z'] )
\newenvironment{vbtflisting}[2][]%
{%
	\noindent
	\begin{minipage}{\textwidth}
		\captionsetup{type=lstlisting}
		\captionof{lstlisting}{#1}#2
		\footnotesize\ttfamily
		}{%
	\end{minipage}
}
