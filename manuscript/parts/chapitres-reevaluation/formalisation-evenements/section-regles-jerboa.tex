Dans la section~\ref{sec:formalisation-evenement-regles-transfo-graphe}, un sommet est créé dans la règle instanciée, ce qui implique qu'il est créé dans la transformation directe. %

Nous allons maintenant montrer que la création du sommet $R\orbit{1,2,3}(n2)$ dans la règle Jerboa détermine la création d'un sommet dans la transformation directe. %

\begin{proposition}[Création d'orbite]\label{prop:formalisation-evenement-regles-jerboa-creation}
	Soit une règle Jerboa $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de
	filtrage $\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte $G$, un type
	d'orbite $\orbit{\omega}$ et $v$ un nœud de $R$.

	L'orbite $\underline{R}\orbit{\omega}(v)$ est créée dans la
	règle si et seulement si toutes les orbites $H\orbit{\omega}(m((v,v')))$ sont créées
	dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous les brins
	$v'$ de l'orbite d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$.
\end{proposition}

\begin{proof}
	Soit $v$ un nœud de $R$ tel que l'orbite $\underline{R}\orbit{\omega}(v)$ est créée
	par la règle, c'est-à-dire tel que
	$\underline{R}\orbit{\omega}(v) \cap \underline{L} = \emptyset$. Soit
	$O = G\orbit{l(h)}(\underline{m}(h))$, l'orbite d'instanciation.

	Soit $w$ un nœud quelconque de l'orbite $\underline{R}\orbit{\omega}(v)$. $w$ est un
	nœud créé de la règle (c'est-à-dire de $R \setminus L$). %
	Donc, par construction de la règle instanciée,
	tous les nœuds $(w,v')$ pour tous les brins $v'$ de l'orbite d'instanciation
	$O$, sont des nœuds créés dans la règle instanciée $r(\underline{m})$.

	Donc toutes les orbites $R(O)\orbit{\omega}((v,v'))$ pour tous les brins $v'$ de
	l'orbite d'instanciation $O$ sont des orbites créées dans la règle instanciée
	$r(\underline{m})$. Et donc, d'après la proposition \ref{prop:base-rule-create},
	toutes les orbites $H\orbit{\omega}(m((v,v')))$ sont créés dans la transformation
	directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous les brins $v'$ de l'orbite
	d'instanciation $O$.
	\\

	Réciproquement, supposons que toutes les orbites $H\orbit{\omega}(m'((v,v')))$ sont créés dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$, pour tous les brins $v'$ de l'orbite d'instanciation $O$.
	Donc, d'après la proposition \ref{prop:base-rule-create}, toutes les orbites $R(O)\orbit{\omega}((v,v'))$ pour tous les brins $v'$ de l'orbite d'instanciation $O$ sont des orbites créées dans la règle instanciée $r(\underline{m})$.
	Donc par construction de l'instanciation, $\underline{R}\orbit{\omega}(v)$ est une orbite créée dans la règle $r$.
\end{proof}

\begin{definition}[Orbite d'origine]
	Soit une règle $r~: L \longrightarrow R$, $n$ un nœud de $R$ et $\orbit{\omega}$ un type d'orbite.

	Une \emph{orbite d'origine} de $\underline{R}\orbit{\omega}(n)$ dans $L$ est une orbite $\underline{L}\orbit{\omega'}(n')$ où :
	\begin{itemize}
		\item où $n'$ est un nœud de $L$,

		\item $\orbit{\omega'}$ est l'ensemble des dimensions  $i$, où $i$ est renommé dans un arc implicite de $n'$ ; c'est-à-dire que $(l_L(n') \mapsto l_R(n))(i) = j \in \orbit{\omega}$

	\end{itemize}
	Soit un morphisme de filtrage $\underline{m} : \underline{L} \rightarrow G$ sur une G-carte $G$, $G \xlongrightarrow{r, \underline{m}} H$ la transformation directe correspondante et $O$ l'orbite d'instanciation. Soit $v'$ un nœud de $O$.

	Une {\em origine} de l'orbite $H\orbit{\omega}((n,v'))$ est une orbite $G\orbit{\omega'}((n',v'))$.
\end{definition}

Prenons un autre exemple de scission avec la règle de décousure de deux volumes
(Figure~\ref{fig:formalisation-evenement-regles-jerboa-regle-decousure-faces}). %
Ici, la face $L\orbit{0,1,3}(n0)$ est scindée en deux dans $R$ puisque le $3$-arc explicite dans $L$ est remplacé par deux boucles dans $R$. %
Cette scission est dite \emph{explicite}, à l'inverse des scission par réécritures
des étiquettes qui sont dites \emph{implicites}.

\begin{proposition}[Scission d'orbite]\label{prop:formalisation-evenement-regles-jerboa-scission}
	Soit une règle $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de
	filtrage $\underline{m}~: \underline{L} \rightarrow R$ sur une G-carte $G$, un type d'orbite
	$\orbit{\omega}$, deux nœuds préservés $v_1$ et $v_2$ de $\underline{K} = \underline{L} \cap \underline{R}$. Notons $O = G\orbit{l(h)}(\underline{m}(h))$ l'orbite d'instanciation.

	$\underline{L}\orbit{\omega}(v_1)$ est scindé dans la règle si :
	\begin{itemize}
		\item la scission est due à des arcs explicites, c'est-à-dire si $\underline{L}\orbit{\omega}(v_1)$ est scindée dans $\underline{R}$~;

		\item la scission est due à des arcs implicites, c'est-à-dire si il existe un nœud $v$ de $\underline{L}\orbit{\omega}(v_1)$ dont l'étiquette $l(v)$ a pour k-ème position une dimension de $\orbit{\omega}$, et que pour tous les nœuds $w$ de $\underline{R}\orbit{\omega}(v_1)$ l'étiquette de $l(w)$ a pour k-ème position est un "$\_$" ou une dimension qui n'appartient pas à $\orbit{\omega}$.
	\end{itemize}

	\begin{enumerate}

		\item
		      Si $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$ dans la règle du à des arcs explicites.
		      Et si l'orbite $\underline{R}\orbit{\omega}(v_1)$ est complète. %
		      Alors, pour tout brin $v'$ de l'orbite d'instanciation $O$, l'orbite $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2, v')))$ dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$ .

		\item Si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites $\underline{R}\orbit{\omega}(v_1)$ et
		      $\underline{R}\orbit{\omega}(v_2)$ dans la règle due à un arc explicite.
		      Si les orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$
		      sont toutes deux incomplètes.
		      Alors, pour chaque brin $v'$ de l'orbite d'instanciation $O$, soit $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites
		      $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2,v')))$, soit l'orbite est modifiée, dans la transformation directe
		      $G \xlongrightarrow{r, \underline{m}} H$.

		\item Si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est scindée dans la règle du au $k$-ième arc implicite.
		      Alors pour tout brin $v'_1$ de l'orbite d'instanciation $O$ et $v'_2$ le $i$-voisin de $v'_1$ dans $O$ avec $i$ la dimension à la $k$-ième position de $l(h)$, $G\orbit{\omega}(m((v_1,v'_1)))$ est soit scindée en deux orbites
		      $H\orbit{\omega}(m'((v_1, v'_1)))$ et $H\orbit{\omega}(m'((v_1,v'_2)))$, soit modifiée en une orbite unique $H\orbit{\omega}(m'((v_1, v'_1))) = H\orbit{\omega}(m((v_1,v'_2)))$ dans la transformation directe
		      $G \xlongrightarrow{r, \underline{m}} H$.

		\item Réciproquement,

		      si l'orbite $G\orbit{\omega}(m((v_1, v'_1))$ est scindée en deux orbites
		      $H\orbit{\omega}(m'((v_1, v'_1)))$ et $H\orbit{\omega}(m'((v_1,v'_2)))$ alors il existe deux n\oe uds préservés $v_3$ et $v_4$ (de $\underline{K} = \underline{L} \cap \underline{R}$) tel que $m((v_3, v'_1)) \in G\orbit{\omega}(m((v_1, v'_1))$ et $m((v_4, v'_1)) \in G\orbit{\omega}(m((v_1, v'_1))$ et $\underline{L}\orbit{\omega}(v_3)$ est scindée en $\underline{R}\orbit{\omega}(v_3)$ et $\underline{R}\orbit{\omega}(v_4)$. %

	\end{enumerate}

\end{proposition}

\begin{proof}
	Soit deux nœuds préservés $v_1$ et $v_2$ de $\underline{K} = \underline{L} \cap \underline{R}$
	\begin{enumerate}
		\item Supposons que $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux
		      orbites $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$, avec au moins l'une des deux complète, dans la
		      règle du à des arcs explicites. Alors, par définition, il existe un chemin
		      $e_1 e_2 ... e_j$ dans $\underline{L}$ de source $v_1$, de cible $v_2$ et étiqueté par
		      $i_1 i_2 ... i_j$ sur $\orbit{\omega}$, mais il n'existe pas de tel chemin
		      $f_1 f_2 ... f_k$ dans $R$ de source $v_1$, de cible $v_2$ et étiqueté sur
		      $\orbit{\omega}$.

		      Donc dans la règle instanciée, pour tout nœud $v'$ de l'orbite
		      d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$, il existe un chemin
		      $(e_1,v') (e_2,v') ... (e_j,v')$ dans le motif gauche de la règle instanciée
		      $L(O)$ dont la source est $(s(e_1),v') = (v_1, v')$, la cible est
		      $(t(e_j), v') = (v_2, v')$ et étiqueté par le même mot $i_1 i_2 ... i_j$ sur
		      $\orbit{\omega}$.

		      Raisonnons par l'absurde. %
		      Supposons que pour un nœud $v'$ de l'orbite d'instanciation $O$, il
		      existe un chemin $(f_1,v') (f_2,v') ... (f_k,v')$ dans le motif droit de la
		      règle instanciée $R(O)$ dont la source est $(s(f_1),v') = (v_1, v')$, la cible
		      est $(t(e_j), v') = (v_2, v')$ et étiqueté sur $\orbit{\omega}$. Alors, par
		      construction de l'instanciation, le chemin $f_1 f_2 ... f_k$ dans $R$ de source
		      $v_1$, de cible $v_2$ et étiqueté sur $\orbit{\omega}$ existe dans la règle de
		      départ. Ce qui est contradictoire avec l'hypothèse. Donc il n'existe pas de
		      chemin $(f_1,v') (f_2,v') ... (f_k,v')$ dans le motif droit de la règle
		      instanciée ayant les propriétés attendues. Donc $R(O)\orbit{\omega}((v_1,v'))$ et
		      $R(O)\orbit{\omega}((v_2,v'))$ sont deux $\orbit{\omega}$-orbites distinctes.

		      Donc d'après la proposition \ref{prop:base-rule_split}, si
		      $\underline{R}\orbit{\omega}(v_1)$ et donc $R(O)\orbit{\omega}((v_1, v'))$ est entièrement
		      filtrée, alors l'orbite $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites
		      $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2, v')))$ dans la
		      transformation directe $G \xlongrightarrow{r,\underline{m}} H$ pour tout nœud
		      $v'$ de l'orbite d'instanciation $O$. %

		\item
		      Supposons que $\underline{L}\orbit{\omega}(v_1)$ est scindée en deux orbites incomplètes
		      $\underline{R}\orbit{\omega}(v_1)$ et $\underline{R}\orbit{\omega}(v_2)$ dans la règle du
		      à des arcs explicites.

		      Pour les mêmes raisons que dans le cas 1, $R(O)\orbit{\omega}((v_1,v'))$ et
		      $R(O)\orbit{\omega}((v_2,v'))$ sont deux $\orbit{\omega}$-orbites distinctes dans la
		      règles instanciée, pour tout nœud $v'$ de l'orbite d'instanciation $O$.

		      Donc d'après la proposition \ref{prop:base-rule_split}, si $\underline{R}\orbit{\omega}(v_1)$
		      et donc $R(O)\orbit{\omega}((v_1, v'))$ est incomplète, alors, pour tout nœud
		      $v'$ de l'orbite d'instanciation $O$, l'orbite
		      $G\orbit{\omega}(m((v_1,v')))$ est soit scindée en deux orbites
		      $H\orbit{\omega}(m'((v_1, v')))$ et $H\orbit{\omega}(m'((v_2, v')))$, soit modifiée dans
		      la transformation directe $G \xlongrightarrow{r,\underline{m}} H$.  %

		\item
		      Supposons que $\underline{L}\orbit{\omega}(v_1)$ est scindée due au $k$-ième arc
		      implicite. Soit $v'_1$ un nœud de l'orbite d'instanciation
		      $O = G\orbit{l(h)}(\underline{m}(h))$ et $v'_2$ le $i$-voisin de $v'_1$ dans
		      $O$, avec $i$ la $k$-ième dimension de l'étiquette $l(h)$ du nœud d'accroche $h$
		      de $L$.

		      Par hypothèse, l'un des nœuds $v$ de $\underline{L}\orbit{\omega}(v_1)$ a une
		      $k$-ième dimension $j$ dans l'$\orbit{\omega}$-orbite ($j \in \orbit{\omega}$), alors que tous
		      des n\oe uds de $\underline{R}\orbit{\omega}(v_1)$ ont leur $k$-ième dimension hors de
		      $\orbit{\omega}$. Donc les nœuds $(v, v'_1)$ et $(v, v'_2)$ sont $j$-voisins dans
		      l'instanciation $L(O)$, donc $(v, v'_1)$ et $(v, v'_2)$ sont dans la même
		      $\orbit{\omega}$-orbite dans $L(O)$, et donc $(v_1, v'_1)$ et $(v_2, v'_2)$ aussi.

		      Nous avons plusieurs cas :
		      \begin{itemize}
			      \item si $(v_1, v'_1)$ et $(v_2, v'_2)$ sont dans deux $\orbit{\omega}$-orbites
			            différentes de $R(O)$, d'après la proposition \ref{prop:base-rule_split}, selon
			            les cas, $H\orbit{\omega}(m'((v_1, v'_1)))$ et $H\orbit{\omega}(m'((v_1,v'_2)))$ sont
			            deux orbites différentes ou non de $H$, avec $H$ le résultat de la
			            transformation directe $G \xlongrightarrow{r, \underline{m}} H$ ; autrement dit
			            $G\orbit{\omega}(m((v_1, v'_1))$ est scindée en $H\orbit{\omega}(m'((v_1, v'_1)))$ et
			            $H\orbit{\omega}(m'((v_1,v'_2)))$ ou juste modifiée par la transformation directe
			            $G \xlongrightarrow{r, \underline{m}} H$ ;

			      \item si au contraire $(v_1, v'_1)$ et $(v_2, v'_2)$ sont dans la même
			            $\orbit{\omega}$-orbites de $R(O)$ (car il existe un chemin $e_1 e_2 ... e_l$ de $O$
			            de source $v'_1$ et de cible $v'_2$ qui ne comprend pas l'étiquette $j$, ce
			            chemin est réécrit en un chemin $(w, e_1) (w,e_2) ... (w, e_l)$ de $R(O)$ dont
			            toutes les étiquettes sont dans $\orbit{\omega}$), d'après la proposition
			            \ref{prop:base-rule-modify}, alors $G\orbit{\omega}(m((v_1, v'_1))$ est modifiée par
			            la transformation directe $G \xlongrightarrow{r, \underline{m}} H$. %
		      \end{itemize}
		\item
		      Soient $m((v_1,v'_1))$ et $m((v_2,v'_2))$ deux brins de $G$ filtrés par la règle avec $v_1$ et $v_2$ deux n\oe uds de $\underline{K} = \underline{L} \cap \underline{R}$ et $v'_1$ et $v'_2$ deux brins de $O$. %
		      Supposons que l'orbite $G\orbit{\omega}(m((v_1,v')))$ est scindée en deux orbites $H\orbit{\omega}(m'((v_1, v'_1)))$ et $H\orbit{\omega}(m'((v_2, v'_2)))$ dans la transformation directe. %
		      D'après la proposition~\ref{prop:base-rule_split}, il existe deux n\oe uds $k_1$ et $k_2$ préservés de la règle instanciée (de $K(O) = L(O) \cap R(O)$) tel que $m(k_1) \in G\orbit{\omega}(m((v_1, v'_1))$ et $m(k_2) \in G\orbit{\omega}(m((v_2,v'_2)$ et $L(O)\orbit{\omega}(k_1)$ est scindée en $R(O)\orbit{\omega}(k_1)$ et $R(O)\orbit{\omega}(k_2)$. %
		      Notons $k_1 = (v_3, v'_3)$ et $k_2 = (v_4, v'_4)$ avec $v_3$ et $v_4$ deux n\oe uds préservés de $\underline{K} = \underline{L} \cap \underline{R}$ et $v'_3$ et $v'_4$ deux brins de $O$. %
		      Nous avons donc $L(O)\orbit{\omega}((v_3,v'_3))$ scindée en $R(O)\orbit{\omega}((v_3, v'_3))$ et $R(O)\orbit{\omega}((v_4, v'_4))$. %
		      Donc il existe au moins chemin dans $L(O)$ entre $(v_3, v'_3)$ et $(v_4, v'_4)$, mais aucun chemin dans $R(O)$ entre $(v_3, v'_3)$ et $(v_4, v'_4)$. %
		      Donc, par construction de l'instanciation $L(O)$, il existe un chemin entre $(v_3, v'_3)$ et $(v_4, v'_3)$ (dont tous les n\oe uds sont de la forme $(v_i, v'_3)$) et un chemin entre $(v_4, v'_3)$ et $(v_4, v'_4)$ (dont tous les n\oe uds sont de la forme $(v_4, v'_i)$). %

		      Raisonnons par l'absurde. %
		      Supposons que la règle Jerboa ne comporte aucune scission. %
		      Par construction de l'instanciation~:
		      \begin{itemize}
			      \item Si la règle Jerboa n'a pas de scission explicite entre $v_3$ et $v_4$, c'est-à-dire que s'il existe un chemin entre $v_3$ et $v_4$ dans $L$ alors il en existe un dans $R$. %
			            Alors, d'après l'instanciation, s'il existe un chemin de $(v_3, v'_3)$ vers $(v_4, v'_3)$ dans $L(O)$, il en existe un dans $R(O)$. %

			      \item Si la règle Jerboa n'a pas de scission implicite entre $\underline{L}\orbit{\omega}(v_3)$ et $\underline{R}\orbit{\omega}(v_3)$. %
			            Alors, d'après l'instanciation, s'il existe un chemin de $(v_4, v'_3)$ vers $(v_4, v'_4)$ dans $L(O)$, il en existe un dans $R(O)$. %
		      \end{itemize}
		      Donc il existe un chemin dans $R(O)$ de $(v_3, v'_3)$ vers $(v_4, v'_4)$, ce qui est contradictoire avec l'hypothèse. %
		      Donc, la règle Jerboa comprend au moins une scission implicite ou explicite. %
	\end{enumerate}
\end{proof}

\begin{proposition}[Fusion d'orbites]
	Soit une règle Jerboa $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de filtrage $\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte $G$, un type d'orbite $\orbit{\omega}$, deux n\oe uds préservés $v_1$ et $v_2$ de $\underline{K} = \underline{L} \cap \underline{R}$ et l'orbite d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$. %

	$\underline{L}\orbit{\omega}(v_1)$ est fusionnée dans la règle~:

	\begin{itemize}
		\item si la fusion est due à des arcs explicites, c'est-à-dire si $\underline{L}\orbit{\omega}(v_1)$ est fusionnées avec $\underline{L}\orbit{\omega}(v_2)$ dans $\underline{R}$~;

		\item si la fusion est due à des arcs implicites, c'est-à-dire si pour tout nœud $v$ de $\underline{L}\orbit{\omega}(v_1)$, l'étiquette $l(v)$ a pour $k$-ème position un "\_" ou une
		      dimension qui n'appartient pas à $\orbit{\omega}$, et qu'il existe un nœud $w$ de $\underline{R}\orbit{\omega}(v_1)$ dont l'étiquette $l(w)$ a pour $k$-ème position une dimension
		      de $\orbit{\omega}$.
	\end{itemize}

	\begin{enumerate}
		\item Si $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées en une orbite $\underline{R}\orbit{\omega}(v_1)$ dans la règle du à des arcs explicites.
		      Et si l'orbite $\underline{L}\orbit{\omega}(v_1)$ est complète. %
		      Alors, pour tout brin $v'$ de $O$, les orbites $G\orbit{\omega}(m((v_1,v')))$ et $G\orbit{\omega}(m((v_2,v')))$ sont fusionnées en une orbite $H\orbit{\omega}(m'((v_1,v')))$ dans la transformation directe
		      $G \xlongrightarrow{r,\underline{m}} H$.

		\item Si les orbites $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont fusionnées en une orbite $\underline{R}\orbit{\omega}(v_1)$ dans la règle du à des arcs explicites.
		      Et si les orbites $\underline{L}\orbit{\omega}(v_1)$ et $\underline{L}\orbit{\omega}(v_2)$ sont toutes deux incomplètes. %
		      Alors, pour chaque brin $v'$ de $O$, soit $G\orbit{\omega}(m((v_1,v')))$ et $G\orbit{\omega}(m((v_2,v')))$ sont fusionnées en une orbite

		      Alors pour tout brin $v_1'$ de $O$ et $v_2'$ le $i$-voisin de $v_1'$ dans $O$ avec $i$ la dimension à la $k$-ème position
		      de $l(h)$, $G\orbit{\omega}(m((v_1,v_1')))$ et $G\orbit{\omega}(m((v_1,v_2')))$ sont soit fusionnées en une orbite $H\orbit{\omega}(m'((v_1,v_1')))$, soit c'est une même orbite modifiée
		      $H\orbit{\omega}(m'((v_1,v_1'))) = H\orbit{\omega}(m'((v_1,v_2')))$ dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$
		\item Réciproquement,

		      si les orbites $L\orbit{\omega}(m((v_1, v'_1))$ et $L\orbit{\omega}(m((v_2, v'_2))$ sont fusionnées en une orbite
		      $H\orbit{\omega}(m'((v_1, v'_1)))$, alors il existe deux n\oe uds préservés $v_3$ et $v_4$ (de $\underline{K} = \underline{L} \cap \underline{R}$) tel que $m((v_3, v'_1)) \in H\orbit{\omega}(m'((v_1, v'_1))$ et $m((v_4, v'_1)) \in H\orbit{\omega}(m'((v_1, v'_1))$ et $\underline{L}\orbit{\omega}(v_3)$ et $\underline{L}\orbit{\omega}(v_4)$ sont fusionnées en  $\underline{R}\orbit{\omega}(v_3) = \underline{R}\orbit{\omega}(v_4) $. %
	\end{enumerate}
\end{proposition}

\begin{proposition}[Non-changement d'orbite]
	Soit une règle Jerboa $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$,
	un morphisme de filtrage $\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte
	$G$, un type d'orbite $\orbit{\omega}$, $v$ un nœud de $L$ et l'orbite d'instanciation $O = G\orbit{l(h)}(\underline{m}(h))$.

	L'orbite $\underline{L}\orbit{\omega}(v)$ est inchangée dans
	la règle si et seulement si, pour tout brin $v'$ de $O$, l'orbite $G\orbit{\omega}(m((v,v')))$ est
	inchangée dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$.
\end{proposition}

\begin{proof}
	Supposons que $\underline{L}\orbit{\omega}(v)$ est inchangée dans la règle Jerboa.

	Soit $w$ un nœud quelconque de l'orbite $\underline{L}\orbit{\omega}(v)$. %
	Donc $w$ est un	nœud inchangé de la règle de $\underline{K} = \underline{L} \cap \underline{R}$. Et donc, par construction de la règle
	instanciée, pour tout brin $v'$ de $O$, le n\oe ud $(w,v')$ est inchangé dans la règle instanciée	$r(\underline{m})$. %
	Or l'orbite $G\orbit{\omega}(m((v,v')))$ n'est pas
	nécessairement complète. %
	Par définition, la partie non filtrée est aussi inchangée. %
	Donc $L(O)\orbit{\omega}((v,v'))$ est inchangée dans la règle instanciée. %
	Et donc, d'après la proposition
	\ref{prop:base-rule-unchange}, pour tout brin $v'$ de $O$, l'orbite $G\orbit{\omega}(m((v,v')))$ est
	inchangée dans la transformation directe $G \xlongrightarrow{r,\underline{m}} H$.

	Réciproquement, supposons que pour tout brin $v'$ de $O$ l'orbite $G\orbit{\omega}(m((v,v')))$ est
	inchangée dans la transformation directe $G \xlongrightarrow{r, \underline{m}} H$. %
	Donc d'après la proposition	\ref{prop:base-rule-unchange}, l'orbite $L(O)\orbit{\omega}((v,v'))$  est inchangée
	dans la règle instanciée. %
	Donc par construction de l'instanciation, l'orbite $\underline{L}\orbit{\omega}(v)$ est inchangée dans la règle Jerboa. %
\end{proof}

\begin{proposition}[Modification d'orbite]
	Soit une règle Jerboa $r~: L \longrightarrow R$, un nœud d'accroche $h$ de $L$, un morphisme de filtrage $\underline{m}~: \underline{L} \rightarrow G$ sur une G-carte $G$, un type d'orbite $\orbit{\omega}$, $v$ un nœud préservé de $\underline{K} = \underline{L} \cap \underline{R}$ et l'orbite d'instanciation	$O = G\orbit{l(h)}(\underline{m}(h))$.

	L'orbite $\underline{L}\orbit{\omega}(v)$ est modifiée dans la règle si et seulement si~:
	\begin{itemize}
		\item si la modification est due à des arcs explicites, alors~:
		      \begin{itemize}
			      \item $\underline{L}\orbit{\omega}(v) \neq \underline{R}\orbit{\omega}(v)$,
			      \item pour tout n\oe ud préservé $w$ de $\underline{K} = \underline{L} \cap \underline{R}$, $w \in \underline{L}\orbit{\omega}(v)$ si et seulement si $w \in \underline{R}\orbit{\omega}(v)$,
			      \item pour toute position d'étiquette $k$, il existe un n\oe ud $w \in \underline{L}\orbit{\omega}(v)$ dont la $k$-ème étiquette de $l(w)$ est une dimension de $\orbit{\omega}$ si et seulement si il existe un n\oe ud $w' \in \underline{R}\orbit{\omega}(v)$ dont la $k$-ème étiquette de $l(w')$ est une dimension de $\orbit{\omega}$~;
		      \end{itemize}
		\item si la modification est due à des arcs implicites, alors~:
		      \begin{itemize}
			      \item $\underline{L}\orbit{\omega}(v) \neq \underline{R}\orbit{\omega}(v)$,
			      \item il existe un n\oe ud préservé $w \in \underline{L}\orbit{\omega}(v)$ tel que l'étiquette de $w$ est différente dans $L$ et $R$,
			      \item pour toute position d'étiquette $k$, il existe un n\oe ud $w \in \underline{L}\orbit{\omega}(v)$ dont la $k$-ème étiquette de $l(w)$ est une dimension de $\orbit{\omega}$ si et seulement si il existe un n\oe ud $w' \in \underline{R}\orbit{\omega}(v)$ dont la $k$-ème étiquette de $l(w')$ est une dimension de $\orbit{\omega}$~;
		      \end{itemize}
	\end{itemize}


	Si l'orbite $\underline{L}\orbit{\omega}(v)$ est modifiée dans la règle Jerboa, alors, pour tout brin $v'$ de $O$,
	$G\orbit{\omega}(m((v,v')))$ est modifiée dans la transformation directe
	$G \xlongrightarrow{r,\underline{m}} H$. %
	Réciproquement, si 	$G\orbit{\omega}(m((v,v')))$ est modifiée dans la transformation directe
	$G \xlongrightarrow{r,\underline{m}} H$, alors $\underline{L}\orbit{\omega}(v)$ est modifiée, scindée ou fusionnée dans la règle. %
\end{proposition}

\begin{definition}[Orbite d'origine]
	Soit une règle $r~: L \longrightarrow R$, $n$ un nœud de $R$ et $\orbit{\omega}$ un type d'orbite.

	Une \emph{orbite d'origine} de $\underline{R}\orbit{\omega}(n)$ dans $L$ est une orbite $\underline{L}\orbit{\omega'}(n')$ où :
	\begin{itemize}
		\item où $n'$ est un nœud de $L$,

		\item $\orbit{\omega'}$ est l'ensemble des dimensions  $i$, où $i$ est renommé dans un arc implicite de $n'$ ; c'est-à-dire que $(l_L(n') \mapsto l_R(n))(i) = j \in \orbit{\omega}$

	\end{itemize}
	Soit un morphisme de filtrage $\underline{m} : \underline{L} \rightarrow G$ sur une G-carte $G$, $G \xlongrightarrow{r, \underline{m}} H$ la transformation directe correspondante et $O$ l'orbite d'instanciation. Soit $v'$ un nœud de $O$.

	Une {\em origine} de l'orbite $H\orbit{\omega}((n,v'))$ est une orbite $G\orbit{\omega'}((n',v'))$.
\end{definition}

Remarquons que lorsqu'une règle n'a pas de membre gauche ($L = \emptyset$), alors toutes les orbites de $R$ sont créées et aucune n'a d'origine.
