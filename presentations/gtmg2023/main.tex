        \begin{block}{Cellule de dimension}
          \begin{itemize}
            \item<2-> 2%

  \begin{block}{Brins et liaisons}<4>
    Les sommets d'une carte généralisée (G-carte) sont appelés \emph{brins}
    et sont reliés par des \emph{liaisons}.
  \end{block}
\end{frame}

      \end{figure}

    \begin{column}{.49\textwidth}
      \begin{block}{Orbite de type}
        \begin{itemize}
          \item<1-> Face $\orbit{\alpha 0, \alpha 1}(a)$%
          \item<2-> Arête $\orbit{\alpha 0, \alpha 2}(a)$%
          \item<3-> Arête $\orbit{\alpha 0, \alpha 2}(c)$%
          \item<4-> Sommet $\orbit{\alpha 1, \alpha 2}(n)$%
          \item<5-> Arête de face $\orbit{\alpha 0}(a)$%
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \begin{block}{Une orbite}<5> Composante connexe issue d'une G-carte $G$, de
    type $\orbit{o}$ et incidente à un brin $b$.
  \end{block}
\end{frame}

\section{Conclusion et perspectives}
\begin{frame}{Conclusion}
  \begin{itemize}
    \item Déduction d'événement à l'aide de G-cartes et de règles de
          transformation de graphes Jerboa
    \item Méthode statique de détection agissant avant l'application des règles
    \item Définition de cas limite
    \item Hors cas limite, aucun calcul dynamique
  \end{itemize}
\end{frame}

\begin{frame}{Perspectives}
  À court terme :
  \begin{itemize}
    \item Définition de cas de tests
    \item Calcul du gain entre le suivi statique et le suivi dynamique
    \item Intégration pour la génération de journaux de bord
    \item Intégration pour le nommage persistant
  \end{itemize}
  \vspace*{1em}
  À plus long terme :
  \begin{itemize}
    \item Détection d'événement dans les scripts de règles
  \end{itemize}
\end{frame}
