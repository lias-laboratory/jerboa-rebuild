\section{Les évènements dans les transformations d'objets}
\label{sec:formalisation-evenements-transfo-objets}

Considérons deux G-cartes $G$ et $H$. %
On veut qualifier les modifications d'orbites de type $\orbit{\omega}$ entre $G$ et $H$.
Ces dernières peuvent être créées, supprimées, scindées, fusionnées, inchangées ou modifiées.
On parle dans ce cas d'événements survenus sur les orbites.

\subsection{Création}
\label{sec:formalisation-evenements-transfo-objets-creation}



\begin{definition}[Création d'orbite]
	Soient $G$ et $H$ deux G-cartes de dimension $n \in \mathbb{N}$,
	$\orbit{\omega}$ un type d'orbite de dimension $n$ et $v$ un brin de $H$.

	L'orbite $H\orbit{\omega}(v)$ est \emph{créée} si et seulement si aucun élément de l'orbite
	$\orbit{\omega}$ incidente au brin $v$ dans $H$ n'existait
	dans $G$, \textit{i.e.} $G \cap H\orbit{\omega}(v) = \emptyset$. %
\end{definition}

\subsection{Scission}
\label{sec:formalisation-evenements-transfo-objets-scission}


\begin{definition}[Scission d'orbite]
	Soient $G$ et $H$ deux G-cartes de dimension $n \in \mathbb{N}$, $\orbit{\omega}$ un type d'orbite de dimension $n$ et $v$ un brin de $G$.

	L'orbite $G\orbit{\omega}(v)$ est \emph{scindée} si et seulement si il
	existe deux brins $v_1$ et $v_2$ de $G\orbit{\omega}(v)$ tels que $v_1$ et
	$v_2$ sont aussi des brins de $H$ et qu'ils appartiennent à des orbites
	$\orbit{\omega}$ différentes, \textit{i.e.}
	$H\orbit{\omega}(v_1) \neq H\orbit{\omega}(v_2)$.
\end{definition}

\subsection{Suppression}
\label{sec:formalisation-evenements-transfo-objets-suppression}


La suppression est l'inverse de la création. %

\begin{definition}[Suppression d'orbite]
	Soient $G$ et $H$ deux G-cartes de dimension $n \in \mathbb{N}$,
	$\orbit{\omega}$ un type d'orbite de dimension $n$ et $v$ un brin de $G$.

	L'orbite $G\orbit{\omega}(v)$ est \emph{supprimée} si et seulement si
	aucun élément de l'orbite $\orbit{\omega}$ incidente à $v$ dans $G$  n'existe dans $H$, \textit{i.e.} $H \cap G\orbit{\omega}(v) = \emptyset$. %
\end{definition}

Reprenons l'exemple de la
Figure~\ref{fig:formalisation-evenements-g-carte-evenement-triangulation-sommet},
cette fois de $H$ vers $G$. %
Le sommet $H\orbit{1,2,3}(a_{2})$ est supprimé dans $G$ car aucun des brins
$a_2,b_2\ldots h_2$ dans $H$ n'existe dans $G$. %


\subsection{Fusion}
\label{sec:formalisation-evenements-transfo-objets-fusion}


La fusion est l'inverse de la scission. %

\begin{definition}[Fusion d'orbites]
	Soient $G$ et $H$ deux G-cartes de dimension $n \in \mathbb{N}$,
	$\orbit{\omega}$ un type d'orbite de dimension $n$ et $v$ et $v'$ deux brins de $G$. %

	Les orbites distinctes $G\orbit{\omega}(v)$ et $G\orbit{\omega}(v')$ sont
	\emph{fusionnées} si et seulement si il existe deux brins respectivement
	$v_{1}$ de $G\orbit{\omega}(v)$ et $v_{2}$ de $G\orbit{\omega}(v')$ tel que
	$v_{1}$ et $v_{2}$ sont des brins de $H$ de la même orbite $\orbit{\omega}$, \textit{i.e.}
	$H\orbit{\omega}(v_{1}) = H\orbit{\omega}(v_{2})$. %

\end{definition}

\subsection{Non-changement}
\label{sec:formalisation-evenements-transfo-objets-non-changement}


\begin{definition}[Non-changement d'orbite]
	Soient $G$ et $H$ deux G-cartes de dimension $n \in \mathbb{N}$,
	$\orbit{\omega}$ un type d'orbite de dimension $n$ et $v$ un brin de $G$.

	L'orbite $G\orbit{\omega}(v)$ est \emph{inchangée} si et seulement si elle identique à $H\orbit{\omega}(v)$, \textit{i.e.} $G\orbit{\omega}(v) = H\orbit{\omega}(v)$.
\end{definition}

Dans la figure~\ref{fig:formalisation-evenements-g-carte-evenement-triangulation-sommet}, l'arête incidente au brin $a_0$ est inchangée, puisqu'elle est identique dans $G$ et $H$. %

\subsection{Modification}
\label{sec:formalisation-evenements-transfo-objets-modification}


\begin{definition}[Modification d'orbite]
	Soient $G$ et $H$ deux G-cartes de dimension $n \in \mathbb{N}$,
	$\orbit{\omega}$ un type d'orbite de dimension $n$ et $v$ un brin de $G$. %

	L'orbite $G\orbit{\omega}(v)$ est \emph{modifiée} si et seulement si il existe un brin $v'$ de
	$G\orbit{\omega}(v)$ et $H$ tel que tout brin préservé (de $G \cap H$) $v''$ appartient à $G\orbit{\omega}(v')$ si et seulement si $v''$ appartient à $H\orbit{\omega}(v')$%
\end{definition}

Dans la
figure~\ref{fig:formalisation-evenements-g-carte-evenement-triangulation-face},
le sommet $G\orbit{1,2,3}(a_{0})$ est modifié dans $H$, sans fusion ni scission, puisque
$G\orbit{1,2,3}(a_{0}) \neq H\orbit{1,2,3}(a_{0})$. %
