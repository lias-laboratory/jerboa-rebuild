\section{Les évènements dans les règles de transformation de graphe}
\label{sec:formalisation-evenement-regles-transfo-graphe}

\maxime{Motiver que l'on vient de caractériser les évènements dans les objets
	dans le cas d'une transformation de $G$ vers $H$ et quand $G$ vers $H$ est une
	transformation directe, on peut caractériser les évènements à partir des seules
	règles (et de leur syntaxe). Si on arrive à suivre les évènements sur les
	règles, alors on les aura pour toutes les applications indépendamment des
	objets sur lesquels on les applique. On pourra ainsi faire le suivi des
	évènements.}

Soient $G$ une G-carte, $r : L \longrightarrow R$ une règle, $m : L \to G$ un morphisme de
filtrage et $H$ la transformation directe $G \xlongrightarrow{(r,m)} H$.
Étudions l'évolution d'une orbite d'intérêt $G\orbit{\omega}(v)$ dans $H$ en
fonction de la règle $r$.

Toutes les définitions des évènements entre deux G-cartes $G$ et $H$ s'étendent
naturellement entre deux motifs $L$ et $R$. %

L'orbite $G\orbit{\omega}(v)$ peut être \emph{entièrement filtrée} par $m$ si
$G\orbit{\omega}(v) \subset m(L)$, \emph{non filtrée} par $m$ si
$G\orbit{\omega}(v) \cap m(L) = \emptyset$, ou \emph{partiellement filtrée} par $m$ dans les
autres cas.

La règle, dans la figure~\ref{fig:formalisation-evenement-regles-transfo-graphe-triangulation}, est la règle de triangulation d'une face
carrée. %
Comme défini dans la section précédente, le sommet $R\orbit{1,2,3}(a_{2})$ est
créé dans $H$. Nous allons montrer que ce sommet est également créé lors de la
transformation directe entre $G$ et $H$.

\begin{proposition}[Création d'orbite]\label{prop:base-rule-create}
	Soient une règle $r : L \longrightarrow R$, un morphisme de filtrage $m : L \rightarrow G$, la
	transformation directe $G \xlongrightarrow{(r,m)} H$, un type d'orbite
	$\orbit{\omega}$ et $v$ un n\oe ud de $R$.

	Une orbite $R\orbit{\omega}(v)$ est créée dans la règle $r$, c'est-à-dire entre L
	et R, si et seulement si $H\orbit{\omega}(m'(v))$ est créée dans la
	transformation directe, c'est-à-dire entre $G$ et $H$.
\end{proposition}

\begin{proof}
	Supposons que l'orbite $R\orbit{\omega}(v)$ est créée dans la règle $r: L \longrightarrow R$.
	D'après la condition d'arcs incidents sur la règle $r$, tout n\oe ud créé de $R$
	et donc tout n\oe ud de l'orbite $R\orbit{\omega}(v)$ est la source d'un $i$-arc, pour
	tout $i$ de $\orbit{\omega}$. %
	Notons que les arcs incidents aux n\oe uds créés sont nécessairement eux-même
	créés. %
	Donc l'orbite $H\orbit{\omega}(m'(v))$ est entièrement filtrée par la règle,
	c'est-à-dire $H\orbit{\omega}(m'(v)) = m'(R\orbit{\omega}(v))$. Et donc
	$H\orbit{\omega}(m'(v))$ est une orbite créée.

	Supposons que l'orbite $H\orbit{\omega}(m'(v))$ est créée dans $H$. Alors par
	définition de la transformation directe, $H\orbit{\omega}(m'(v))$ est l'image de
	l'orbite $R\orbit{\omega}(v)$ par $m'$. Et donc $R\orbit{\omega}(v)$ est une orbite
	créée par la règle $r$.
\end{proof}

\begin{proposition}
	\label{prop:base-rule-delete}
	Soit une règle $r : L \longrightarrow R$, un morphisme de filtrage $m : L \rightarrow G$, la transformation directe $G \xlongrightarrow{(r,m)} H$, un type d'orbite $\orbit{\omega}$ et $v$ un n\oe ud de $L$.

	Une orbite $L\orbit{\omega}(v)$ est \emph{supprimée} dans la règle $r$,
	c'est-à-dire entre $L$ et $R$, si est seulement si $G\orbit{\omega}(m(v))$ est \emph{supprimée} dans
	la transformation directe, c'est-à-dire entre $G$ et $H$.
\end{proposition}

\begin{proof}
	Supposons que l'orbite $L\orbit{\omega}(v)$ est supprimée dans la règle $r : L \longrightarrow R$.
	D'après la condition d'arcs incidents sur la règle $r$, tout n\oe ud supprimé de $L$ et donc tout
	n\oe ud de l'orbite $L\orbit{\omega}(v)$ est la source d'un $i$-arc, pour tout $i$ de $\orbit{\omega}$.
	Donc l'orbite $G\orbit{\omega}(m(v))$ est entièrement filtrée par la règle, c'est-à-dire
	$G\orbit{\omega}(m(v)) =  m(L\orbit{\omega}(v))$. Et donc $G\orbit{\omega}(m(v))$ est une orbite supprimée.

	Supposons que l'orbite $G\orbit{\omega}(m(v))$ est supprimée dans $H$. Alors par définition de la
	transformation directe, $G\orbit{\omega}(m(v))$ est l'image de l'orbite $L\orbit{\omega}(v)$ par $m$.
	Et donc $L\orbit{\omega}(v)$ est une orbite supprimée par la règle $r$.
\end{proof}

\begin{proposition}[Non-changement d'orbite]
	\label{prop:base-rule-unchange}
	Soit une règle $r : L \longrightarrow R$, un morphisme de filtrage $m : L \rightarrow G$, la transformation directe $G \xlongrightarrow{(r,m)} H$, un type d'orbite $\orbit{\omega}$,  $v$ un n\oe ud de $K = L \cap R$.

	Une orbite $L\orbit{\omega}(v)$ est \emph{inchangée}
	dans la règle $r$, c'est-à-dire entre $L$ et $R$, si et seulement si $G\orbit{\omega}(m(v))$ est
	\emph{inchangée} dans la transformation directe, c'est-à-dire entre $G$ et $H$.
\end{proposition}

\begin{proof}
	Supposons que l'orbite $L\orbit{\omega}(v)$ est inchangée dans la règle $r : L \longrightarrow R$.
	L'orbite $G\orbit{\omega}(m(v))$ n'est pas nécessairement entièrement filtrée,
	c'est-à-dire
	$m(L\orbit{\omega}(v)) \subset G\orbit{\omega}(m(v))$ et $m'(R\orbit{\omega}(v)) \subset H\orbit{\omega}(m'(v))$.
	Par définition de la transformation directe, la partie filtrée de l'orbite est inchangée~: $m(L\orbit{\omega}(v)) = m'(R\orbit{\omega}(v))$. Par ailleurs, la partie non filtrée est aussi inchangée, \textit{i.e.} $G\orbit{\omega}(m(v)) \setminus m(L\orbit{\omega}(v)) = H\orbit{\omega}(m'(v)) \setminus m'(R\orbit{\omega}(v))$.
	Donc $G\orbit{\omega}(m(v))$ est une orbite inchangée.

	Supposons que l'orbite $G\orbit{\omega}(m(v))$ est inchangée entre $G$ et $H$, c'est-à-dire que $G\orbit{\omega}(m(v)) = H\orbit{\omega}(m'(v))$.
	Alors par définition de la
	transformation directe, $L\orbit{\omega}(v)$ est une orbite inchangée par la règle $r$.

\end{proof}

Remarquons qu'en plus des orbites (partiellement) filtrées, les orbites de $G$ non filtrées (telles que $G\orbit{\omega}(v) \cap m(L) = \emptyset$) sont par définition inchangées. On dira que la règle est \emph{sans effet} sur l'orbite.

\begin{proposition}[Scission d'orbite]
	\label{prop:base-rule_split}
	Soit une règle $r~: L \longrightarrow R$, un morphisme de filtrage $m : L \rightarrow G$, la transformation directe $G \xlongrightarrow{(r,m)} H$, un type d'orbite $\orbit{\omega}$ et deux n\oe uds préservés $v_1$ et $v_2$ de $K = L \cap R$.

	\begin{enumerate}
		\item Si l'orbite $R\orbit{\omega}(v_1)$ est entièrement filtrée, c'est-à-dire que pour tout n\oe ud $v$ de $R\orbit{\omega}(v_1)$ et toute dimension $i$ de $\orbit{\omega}$, $R\orbit{\omega}(v_1)$ possède un arc de dimension $i$ et de source~$v$,

		      si l'orbite $L\orbit{\omega}(v_1)$ est scindée en deux orbites $R\orbit{\omega}(v_1)$ et $R\orbit{\omega}(v_2)$ dans la règle, alors $G\orbit{\omega}(m(v_1))$ est scindée en deux orbites $H\orbit{\omega}(m'(v_1))$ et $H\orbit{\omega}(m'(v_2))$ dans la transformation directe.

		\item Si les orbites $R\orbit{\omega}(v_1)$ et $R\orbit{\omega}(v_2)$ sont partiellement filtrées,

		      si l'orbite $L\orbit{\omega}(v_1)$ est scindée en deux orbites $R\orbit{\omega}(v_1)$ et $R\orbit{\omega}(v_2)$ dans la règle, alors $G\orbit{\omega}(m(v_1))$ est soit scindée en deux orbites $H\orbit{\omega}(m'(v_1))$ et $H\orbit{\omega}(m'(v_2))$ dans la transformation directe, soit modifiée (et alors $H\orbit{\omega}~(m'(v_1)) = H\orbit{\omega}(m'(v_2))$.

		\item Réciproquement,

		      si l'orbite $G\orbit{\omega}(m(v_1))$ est scindée en deux orbites $H\orbit{\omega}(m'(v_1))$ et $H\orbit{\omega}~(m'(v_2))$ dans la transformation directe, alors l'orbite $L\orbit{\omega}(v_1)$ est scindée en deux orbites $R\orbit{\omega}(v_1)$ et $R\orbit{\omega}(v_2)$ dans la règle.
	\end{enumerate}
\end{proposition}

%

\begin{proof}

	Point 1 :

	Supposons que l'orbite $L\orbit{\omega}(v_1)$ est scindée dans la règle $r : L \longrightarrow R$ avec l'orbite $R\orbit{\omega}(v_1)$ entièrement filtrée. L'orbite $G\orbit{\omega}(m(v_1))$ n'est pas nécessairement
	entièrement filtrée, c'est-à-dire que $m(L\orbit{\omega}(v_1)) \subset G\orbit{\omega}(m(v_1))$. De même, $m'(R\orbit{\omega}(v_2)) \subset H\orbit{\omega}(m'(v_2))$.
	Par hypothèse, l'orbite $R\orbit{\omega}(v_1)$ est entièrement filtrée, donc $H\orbit{\omega}(m'(v_1)) = m'(R\orbit{\omega}(v_1))$, et est distincte de $R\orbit{\omega}(v_2))$, donc $m'(R\orbit{\omega}(v_1)) \cap m'(R\orbit{\omega}(v_2)) = \emptyset$.
	Donc $H\orbit{\omega}(m'(v_1)) \cap m'(R\orbit{\omega}(v_2)) = \emptyset$.
	Or $m'(v_2)$ n'est pas un n\oe ud de $H\orbit{\omega}(m'(v_1))$, donc $H\orbit{\omega}(m'(v_1))$ et $H\orbit{\omega}(m'(v_2))$ sont des orbites distinctes.
	Et donc $G\orbit{\omega}(m(v_1))$ est une orbite
	scindée.

	Point 2 :

	Supposons que l'orbite $L\orbit{\omega}(v_1)$ est scindée dans la règle $r : L \longrightarrow R$ avec $R\orbit{\omega}(v_1)$ et $R\orbit{\omega}(v_2)$ toutes deux partiellement filtrées.

	Si $m'(v_1)$ et $m'(v_2)$ appartiennent à la même $\orbit{\omega}$-orbite dans $H$, c'est-à-dire
	$H\orbit{\omega}(m'(v_1)) = H\orbit{\omega}(m'(v_2))$, alors l'orbite $G\orbit{\omega}(m(v_1))$ est modifiée dans la transformation directe.

	Si $m'(v_1)$ et $m'(v_2)$ appartiennent à deux $\orbit{\omega}$-orbites  dans $H$, c'est-à-dire
	$H\orbit{\omega}(m'(v_1)) \cap H\orbit{\omega}(m'(v_2)) = \emptyset$, alors l'orbite $G\orbit{\omega}(m(v_1))$ est scindée dans la transformation directe.

	Point 3 :

	Supposons que l'orbite $G\orbit{\omega}(m(v_1))$ est scindée dans $H$ en $H\orbit{\omega}(m'(v_1))$ et $H\orbit{\omega}(m'(v_2))$, c'est-à-dire que $G\orbit{\omega}(m(v_1)) = G\orbit{\omega}(m(v_2))$ et $H\orbit{\omega}(m'(v_1)) \neq H\orbit{\omega}(m'(v_2))$. %
	Résonnons par l'absurde. %
	Supposons que pour tout couple de n\oe uds préservés $k_1$ et $k_2$ de $K = L \cap R$, $L\orbit{\omega}(k_1)$ n'est pas scindée en $R\orbit{\omega}(k_1)$ et $R\orbit{\omega}(k_2)$.
	Alors pour tout chemin $c$ dans $G$ de $m(v_1)$ à $m(v_2)$ étiqueté dans $\orbit{\omega}^*$ ( \textit{i.e.}, $c$ est un chemin de $G\orbit{\omega}(m(v_1))$), $c$ n'existe pas dans $H$. %
	Donc par définition de la transformation directe, il existe un sous-chemin filtré $m(c')$ de $c$ tel que $c'$ est un chemin de $L$ mais pas de $R$. %

	Alors par définition de la transformation directe, $R\orbit{\omega}(v_1) \neq R\orbit{\omega}(v_2)$. Supposons que $L\orbit{\omega}(v_1) \neq L\orbit{\omega}(v_2)$ \todo{reprendre ici: ne pas rentrer trop dans le détail, on ne veut pas énumérer une infinité de chemins}

	est une orbite scindée par la règle $r$.
\end{proof}

Si les deux orbites scindées ne sont ni l'une ni l'autre entièrement filtrées par la règle, alors selon l'objet transformé, elles peuvent filtrer des orbites différentes ou une seule et même orbite. Donc selon les cas, la scission peut exister ou non dans l'objet.

\begin{proposition}[Fusion d'orbites]
	\label{prop:base-rule-merge}
	Soit une règle $r : L \longrightarrow R$, un morphisme de filtrage $m : L \rightarrow G$, la transformation directe $G \xlongrightarrow{(r,m)} H$, et un type d'orbite $\orbit{\omega}$.

	\begin{enumerate}
		\item Soit deux n\oe uds préservés $v_1$ et $v_2$ de $L$ et $R$ tels que l'orbite $L\orbit{\omega}(v_1)$ est entièrement
		      filtrée, c'est-à-dire que pour tout n\oe ud $v$ de $L\orbit{\omega}(v_1)$ et toute dimension $i$ de $\orbit{\omega}$,
		      $L\orbit{\omega}(v_1)$ possède un arc de dimension $i$ et de source~$v$.

		      Si les orbites $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont fusionnées en une orbite $R\orbit{\omega}(v_1)$ dans la règle,
		      alors $G\orbit{\omega}(m(v_1))$ et $G\orbit{\omega}(m(v_2))$ sont fusionnées en une orbite $H\orbit{\omega}(m'(v_1))$ dans la
		      transformation directe.

		\item Soit deux n\oe uds préservés $v_1$ et $v_2$ de $L$ et $R$ tels que les orbites $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont partiellement
		      filtrées.

		      Si les orbites $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont fusionnées en une orbite $R\orbit{\omega}(v_1)$ dans la règle,
		      alors $G\orbit{\omega}(m(v_1))$ et $G\orbit{\omega}(m(v_2))$ sont soit fusionnées en une orbite $H\orbit{\omega}(m'(v_1))$ dans
		      la transformation directe, soit modifiées et alors $G\orbit{\omega}(m(v_1)) = G\orbit{\omega}(m(v_2))$.

		\item Soit deux n\oe uds préservés $v_1$ et $v_2$ de $L$ et $R$.

		      Si les orbites $G\orbit{\omega}(m(v_1))$ et $G\orbit{\omega}(m(v_2))$ sont fusionnées en une orbite $H\orbit{\omega}(m'(v_1))$ dans
		      la transformation directe, alors les orbites $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont fusionnées en une orbite
		      $R\orbit{\omega}(v_1)$ dans la règle.
	\end{enumerate}
\end{proposition}

\begin{proof}

	Point 1 :

	Supposons que les orbites $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont fusionnées dans la règle $r : L \longrightarrow R$ avec l'orbite $L\orbit{\omega}(v_1)$ entièrement filtrée. L'orbite $H\orbit{\omega}(m'(v_1))$ n'est pas nécessairement
	entièrement filtrée, c'est-à-dire que $m'(R\orbit{\omega}(v_1)) \subset H\orbit{\omega}(m'(v_1))$. De même, $m(L\orbit{\omega}(v_2)) \subset G\orbit{\omega}(m(v_2))$.
	Par hypothèse, l'orbite $L\orbit{\omega}(v_1)$ est entièrement filtrée, donc $G\orbit{\omega}(m(v_1))= m(L\orbit{\omega}(v_1))$, et est distincte de $L\orbit{\omega}(v_2))$, donc $m(L\orbit{\omega}(v_1)) \cap m(L\orbit{\omega}(v_2)) = \emptyset$.
	Donc $G\orbit{\omega}(m(v_1)) \cap m(L\orbit{\omega}(v_2)) = \emptyset$.
	Or si $m(v_2)$ n'est pas un n\oe ud de $G\orbit{\omega}(m(v_1))$, alors $G\orbit{\omega}(m(v_1))$ et $G\orbit{\omega}(m(v_2))$ sont des orbites distinctes.
	Et donc $H\orbit{\omega}(m'(v_1))$ est une orbite scindée.

	Point 2 :

	Supposons que les orbites $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont fusionnées dans la règle $r : L \longrightarrow R$ avec $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ toutes deux partiellement filtrées.

	Si $m(v_1)$ et $m(v_2)$ appartiennent à la même $\orbit{\omega}$-orbite dans $G$, c'est-à-dire
	$G\orbit{\omega}(m(v_1)) = G\orbit{\omega}(m(v_2))$, alors l'orbite $H\orbit{\omega}(m'(v_1))$ est modifiée dans la transformation directe.

	Si $m(v_1)$ et $m(v_2)$ appartiennent à deux $\orbit{\omega}$-orbites dans $G$, c'est-à-dire
	$G\orbit{\omega}(m(v_1)) \cap G\orbit{\omega}(m(v_2)) = \emptyset$, alors les orbites $G\orbit{\omega}(m(v_1))$
	et $G\orbit{\omega}(m(v_2))$ sont fusionnées dans la transformation directe.

	Point 3 :

	Supposons que les orbites $G\orbit{\omega}(m(v_1))$ et $G\orbit{\omega}(m(v_2))$ sont fusionnées dans $H$, c'est-à-dire que $G\orbit{\omega}(m(v_1)) \cup G\orbit{\omega}(m(v_2)) \subset H\orbit{\omega}(m'(v_1))$.
	Alors par définition de la transformation directe, $L\orbit{\omega}(v_1)$ et $L\orbit{\omega}(v_2)$ sont des orbites fusionnées par la règle $r$.
\end{proof}

\subsection{Modification}

Si une orbite est modifiée dans la règle, alors elle l'est dans l'objet. En particulier, une modification dans la règle, même partiellement filtrée, ne peut pas produire une scission ou une fusion dans l'objet transformé.

\begin{proposition}[Modification d'orbite]
	\label{prop:base-rule-modify}
	Soit une règle $r : L \longrightarrow R$, un morphisme de filtrage $m : L \rightarrow G$, la transformation directe $G \xlongrightarrow{(r,m)} H$, un type d'orbite $\orbit{\omega}$ et $v$ un n\oe ud préservé de $K = L \cap R$.

	Si une orbite $L\orbit{\omega}(v)$ est modifiée
	dans la règle $r$, alors $G\orbit{\omega}(m(v))$ est
	modifiée dans la transformation directe.

	Réciproquement, si une orbite $G\orbit{\omega}(m(v))$ est modifiée dans la transformation directe, alors $L\orbit{\omega}(v)$ est modifiée, fusionnée ou scindée dans la règle.
\end{proposition}

\begin{proof}
	Soit $v$ un n\oe ud préservé de $K = L \cap R$ tel que l'orbite $L\orbit{\omega}(v)$ est modifiée dans la règle $r$.
	Donc par définition :
	\begin{itemize}
		\item $L\orbit{\omega}(v) \neq R\orbit{\omega}(v)$ ; et donc par transformation directe, $G\orbit{\omega}(m(v)) \neq H\orbit{\omega}(m'(v))$ ;

		\item pour tout n\oe ud préservé $v'$ de $K = L \cap R$, $v'$ est un n\oe ud de $L\orbit{\omega}(v)$ si et seulement si c'est un n\oe ud de $R\orbit{\omega}(v)$~; donc $m(v')$ est un n\oe ud de $G\orbit{\omega}(m(v))$ si et seulement si c'est un n\oe ud de $H\orbit{\omega}(m'(v))$~; par ailleurs, les n\oe uds non filtrés de $G\orbit{\omega}(m(v))$ sont nécessairement des n\oe uds non filtrés dans $H\orbit{\omega}(m'(v))$ ; par conséquent, pour tout n\oe ud $v^{\prime\prime}$ conservé de $G \cap H$, $v^{\prime\prime}$ est un n\oe ud de $G\orbit{\omega}(m(v))$ si et seulement si c'est un n\oe ud de $H\orbit{\omega}(m'(v))$~;
	\end{itemize}

	De manière réciproque, soit $v$ un n\oe ud conservé de $K = L\cap R$ tel que l'orbite $G\orbit{\omega}(m(v))$ est modifiée dans la transformation directe $G \xlongrightarrow{r,m} H$.

	Par définition $G\orbit{\omega}(m(v)) \neq H\orbit{\omega}(m'(v))$ et pour tout brin $v'$ de $G \cup H$, $v'$ appartient à $G\orbit{\omega}(m(v))$ si et seulement si il appartient à $H\orbit{\omega}(m'(v))$. %
	Soit $v'$ est filtré, c'est-à-dire qu'il existe un n\oe ud $w$ de $K$ tel que $m(w) = v'$. Dans ce cas  \todo{a reprendre}
	% Or d'après les propositions précédentes, il est impossible que $L\orbit{\omega}(v)$ soit inchangée ou supprimée dans la règle $r$, de même $R\orbit{\omega}(v)$ ne peut pas être créée dans $r$. En revanche, $L\orbit{\omega}(v)$ peut être scindée, fusionnée ou modifiée dans la règle~$r$.
\end{proof}
