\begin{definition}[Graphe étiqueté]
	\label{def:graphe-etiquete}
	Soient deux ensembles d'étiquetage $\mathcal{C}_{V}$ pour les nœuds et $\mathcal{C}_{E}$ pour
	les arcs. %

	Un \emph{graphe étiqueté} $G=(V,E,s,t,l_{V},l_{E})$ est défini par~:
	\begin{itemize}
		\item un ensemble $V$ de \emph{nœuds}~;
		\item un ensemble $E$ d'\emph{arcs}~;
		\item deux fonctions \emph{source} $s : E \to V$ et \emph{cible} $t : E \to V$
		      qui lient les arcs aux nœuds~;
		\item deux \emph{fonctions d'étiquetage} $l_{V} : V -> \mathcal{C}_{V}$ et
		      $l_{E} : E \to \mathcal{C}_{E}$ qui étiquettent respectivement les nœuds et les
		      arcs.%
	\end{itemize}
	Soient un nœud $v \in V$ et un arc $e \in E$~; $s(e)$, $t(e)$ sont appelés
	respectivement le nœud source et le nœud cible de $e$ et $l_{V}(v)$ et
	$l_{E}(e)$ sont respectivement les étiquettes de $v$ et $e$.
\end{definition}

\begin{definition}[Morphisme]
	\label{def:morphismes-graphes}
	Soient $G = (V,E,s,t,l_{V}, l_{E})$
	et $G' = (V',E',s',$ $t',l_{V}', l_{E}')$ deux graphes étiquetés.
	%et $m : G \to G'$ un morphisme.

	Un \emph{morphisme} $m: G \to G'$ est un couple de fonctions
	$m_{V} : V \to V'$, $m_{E} : E \to E'$ préservant les sources, les cibles et
	les étiquettes de G dans G', c'est-à-dire pour tout arc $e \in E$,
	$m_{V}(s(e)) = s'(m_{E}(e))$, $m_{V}(t(e)) = t'(m_{E}(e))$, $l_{V}(e) = l_{V}'(m_{V}(v))$ et
	$l_{E}(e) = l_{E}'(m_{E}(e))$.

	\noindent $m$ est~:
	\begin{itemize}
		\item \emph{injectif} si ses fonctions $m_{V}$ et $m_{E}$ sont
		      injectives\footnotemark,
		\item un \emph{isomorphisme} si ses fonctions $m_{V}$ et $m_{E}$ sont des
		      bijections\footnotemark,
		\item une \emph{inclusion}, noté $\iota : G \hookrightarrow G'$ ou simplement $G \hookrightarrow G'$, si
		      $\iota_{V}$ et $\iota_{E}$ sont les applications identités.
	\end{itemize}
\end{definition}

\footnotetext{Pour rappel, une fonction
	$f : A \to B$ est injective si et seulement si tout élément $b \in B$ à au plus un
	antécédant par $f$, c'est-à-dire $\forall a,a' \in A^{2}$, si $f(a) = f(a')$ alors
	$a = a'$} %
%
\stepcounter{footnote}%
\footnotetext{Pour rappel, une fonction $f : A \to B$ est

	antécédant par $f$, c'est-à-dire $\forall b \in B$, $\exists!$ $a \in A$ tel que $f(a) =
		b$} %

À l'aide de la notion de morphisme, il est possible d'étendre les principales
notions ensemblistes aux graphes. %
Ces notions ensemblistes sont, entre autres, l'union, l'intersection ou encore
la différence entre deux graphes. %

\begin{definition}[Inclusion, union (disjointe), intersection, différence]
	\label{def:graphe-notions-ensemblistes}
	Soient $G = (V,E,s,t,l_{V},l_{E})$ et $G'=(V',E',s',t',l'_{V},l'_{E})$ deux
	graphes étiquetés. %

	\begin{itemize}
		\item $G$ est  \emph{inclus} dans $G'$, si et seulement si il existe un morphisme d'inclusion $G \hookrightarrow G'$.

		\item L'\emph{intersection} $G \cap G'$, quand elle existe, est le graphe
		      $K$ dont les nœuds sont $V \cap V'$ et les arcs sont $E \cap E'$, tel
		      que $G \hookleftarrow K \hookrightarrow G'$.

		\item L'\emph{union} $G \cup G'$, quand elle existe, est le graphe $J$ sont
		      les nœuds sont $V \cup V'$ et les arcs sont $E \cup E'$, tel que
		      $G \hookrightarrow K \hookleftarrow G'$.

		\item L'\emph{union disjointe} $G \sqcup G'$ est l'union de deux graphes
		      disjoints $G$ et $G'$ (tels que $V \cap V' = \emptyset$ et
		      $E \cap E' = \emptyset$)
		\item La \emph{différence} $G \setminus G'$ est la structure
		      $(V \setminus V', E \setminus E', s_{|E \setminus E'}, t_{|E \setminus E'}, l_{E_{|E \setminus E'}})$.

		\item Soit $\emptyset$ le \emph{graphe vide}, c'est-à-dire l'unique graphe
		      ayant un ensemble vide de nœuds et d'arcs.
	\end{itemize}
\end{definition}

\begin{definition}[Sous-graphe]
	\label{def:sous-graphe}
	Soient deux graphes $G=(V,E,s,t,l_{V},l_{E})$ et
	$G'=(V',E',s',t',l'_{V},l'_{E})$, $\omega$ un mot sur $l_{E}$
	et $v \in V$. %

	$G'$ est un \emph{sous-graphe} de $G$ si et seulement si, $G' \subset G$.
	On note $G\orbit{\omega}(v)$ le sous-graphe qui contient le nœud $v$, tous les nœuds atteignables par des arcs étiquetés par des lettres de $\omega$ et ces arcs eux-mêmes. %

\end{definition}

Un dernier concept essentiel pour l'utilisation des graphes et celui du chemin. %
Intuitivement, un chemin est une suite d'arcs traversée, à la suite, lors d'un
parcours depuis un nœud $v_{i}$ jusqu'à un nœud $v_{k}$. %

\begin{definition}[Chemin et cycle]
	\label{def:chemin-et-cycle}
	Un \emph{chemin} $e_{1}e_{2} \dots e_{k}$ dans un graphe $G$ est
	une suite d'arcs tels que pour $1 \leq i < k$, $t(e_{i}) = s(e_{i+1})$.
	La source d'un tel chemin est $s(e_{1})$, sa cible est $t(e_{k})$ et son
	étiquette est le mot $\alpha(e_{1}) \alpha(e_{2}) \dots \alpha(e_{k})$.

	\noindent Un \emph{cycle} est un chemin $e_1,e_2,\dots e_k$ pour lequel la
	source est également la cible du dernier arc, \textit{i.e.} $s(e_{1}) = t(e_{k})$.
\end{definition}

\begin{definition}[Règle de transformation de graphe]
	\label{def:regle-transformation-graphe}
	Soient deux graphes étiquetés $L$ et $R$ tels que l'intersection
	$L \cap R = K$ existe. %

	$r : L \longrightarrow R$ est la \emph{règle de transformation de graphe}
	définie par le double morphisme d'inclusion
	$L \hookleftarrow K \hookrightarrow R$.
	$L$ est appelé le membre gauche, ou motif gauche, $R$ le membre droit, ou motif droit, et $K$ l'interface.
\end{definition}

Une telle règle est illustrée dans la
Figure~\ref{fig:etat-art-model-regle-regle-transformation-graphe} où le membre
gauche ($L$) filtre tout ou partie d'un graphe et le membre droit ($R$)
transforme l'élément filtré par le membre gauche. %
Une manière simple d'assurer que l'intersection $L \cap R$ est un graphe est que
$L$ et $R$ n'aient pas d'arête commune. %

L'application d'une règle peut être définie formellement dans la théorie des
catégories~\cite{ehrig2006fundamentals}. %
Dans ce manuscrit, nous en donnons une définition ensembliste équivalente.
L'application de la règle $r: L \longrightarrow R$ sur un graphe $G$ le long d'un morphisme
de filtrage $m: L \rightarrow G$ produit un graphe transformé $H$, où $m'$ est un
morphisme injectif qui étend $m$ sur $R$. %
Intuitivement, $m(L \setminus R)$ est supprimé dans $G$, puis
$m'(R \setminus L)$ est ajouté pour produire la transformation directe~$H$. %

\begin{definition}[Application d'une règle de transformation de graphe]
	\label{def:application-regle-transformation-graphe}
	Soient $r: L \longrightarrow R$ une règle d'interface $K$ et
	$m : L \rightarrow G$ un morphisme injectif appelé morphisme de filtrage. %

	%
	L'\emph{application} $G \xlongrightarrow{(r, m)} H$ de $r$ sur $G$ le long
	de $m$, lorsqu'elle existe, est définie par le double morphisme d'inclusion
	$G \hookleftarrow D \hookrightarrow H$, où
	$D = (G \setminus m(L \setminus K))$ et $H = D \sqcup m'(R \setminus K)$,
	avec $m'$ une restriction de $m$ sur $K$, puis son extension sur $R$ telle
	que $m'$ soit injective. %
	$m'$, et donc $H$, est unique à isomorphisme près. %
	$H$ est la transformation directe de $G$ par la règle $r$ le long du
	morphisme de filtrage $m$. %
\end{definition}

\subsection[Les cartes généralisées]{Les cartes généralisées}
\label{sec:etat-art-model-regles-gcartes}


Les \emph{cartes généralisées} (ou «~G-cartes~»)
\cite{lienhardt1991topological,Damiand-Lienhardt14} permettent la représentation
de variétés géométriques (avec ou sans bord) basées sur des structures
topologiques cellulaires de dimension quelconque.

\begin{definition}[G-carte]\label{def:cartes-generalisees}%
	\label{def:gcarte}
	Un \emph{graphe topologique} de dimension $n \in \mathbb{N}$ est un graphe étiqueté
	$G = (V,E,s,t,\alpha)$, où les nœuds sont appelés brins, les arcs sont appelés liaisons et $\alpha : E \to [0,n]$ est la fonction d'étiquetage des liaisons.

	Une \emph{G-carte} est un graphe topologique
	% $G = (V,E,s,t,\alpha)$, où les nœuds sont appelés brins et les arcs sont appelés liaisons et $\alpha : E \to [0,n]$ est la fonction d'étiquetage des liaisons,
	qui vérifie les contraintes suivantes~:
	\begin{itemize}
		\item \emph{non-orientation}~: pour tout arc $e \in E$, il existe un arc
		      inverse $e' \in E$ tel que $s(e') = t(e)$, $t(e') = s(e)$ et
		      $\alpha(e') = \alpha(e)$~;
		\item \emph{arcs incidents}~: pour toute dimension $i \in [0,n]$ et pour
		      tout nœud $v \in V$, il existe exactement un arc $e \in E$ d'étiquette
		      $\alpha(e)=i$ et de source $s(e)=v$~;
		      % \xavier{C'est ici qu'on peut écrire que les G-cartes sont étiquetées par des dimensions topologiques.}
		\item \emph{cycle}~: pour toutes dimensions $i, j \in [0,n]^{2}$ telles que
		      $i+2 \leq j$, et pour tout nœud $v \in V$, il existe un cycle, étiqueté
		      par \emph{$ijij$}, de source $v$.
	\end{itemize}
\end{definition}

\subsubsection[Règles de transformation de graphe]{Règles de transformation de graphe}
\label{sec:etat-art-model-regles-lang-jerboa-regles-inst}

% Les règles de transformation de G-cartes sont les règles instanciées de Jerboa.

Les règles de transformation de graphe (définition~\ref{def:regle-transformation-graphe}) permettent de construire des objets représentés par des G-cartes.

Des conditions syntaxiques sur les règles définies dans
\cite{poudr2009,pascual2022topological} permettent de garantir la préservation
des G-cartes. %
% Jerboa et son éditeur vérifient automatiquement que les règles éditées respectent ces conditions syntaxiques. %

\begin{theoreme}[Préservation de la cohérence topologique]
	\label{thm:preservation-coherence-topo-gcarte}
	Soit $r : L \longrightarrow R$ une règle de transformation de graphe topologique de dimension $n \in \mathbb{N}$. Si $r$
	vérifie les propriétés syntaxiques suivantes, alors pour tout morphisme de
	filtrage $m : L \to G$ d'une G-carte $G$, la transformation directe
	$G \xlongrightarrow{r,m} H$ produit une G-carte $H$~:


	\begin{itemize}
		\item \emph{Condition de non-orientation}~: $L$ et $R$ sont
		      \emph{non-orientés}.
		\item \emph{Conditions d'arcs incidents}~: pour toute dimension
		      $i \in [0,n]$,
		      \begin{itemize}
			      \item un nœud préservé de $L \cap R$ est la source d'un arc
			            étiqueté $i$ dans $L$ si et seulement si il est la source
			            d'un arc étiqueté $i$ dans $R$~;
			      \item tout nœud supprimé de $L \setminus R$ et tout nœud créé de $R \setminus L$
			            est la source d'un unique arc étiqueté $i$.
		      \end{itemize}
		\item \emph{Conditions de cycles}~: pour tout couple $(i,j)$ tel que
		      $0 \leq i \leq i + 2 \leq j \leq n$,
		      \begin{itemize}
			      \item un nœud créé de $R \setminus L$ est la source d'un cycle étiqueté
			            $ijij$;
			      \item si un nœud préservé de $L \cap R$ est source d'un cycle
			            étiqueté $ijij$ dans $R$, il est source d'un cycle
			            étiqueté $ijij$ dans $L$~;
			      \item si un nœud préservé de $L \cap R$ n'est pas la source d'un
			            cycle étiqueté $ijij$ dans $L$, alors les arcs
			            d'étiquettes $i$ et $j$, dont il est la source, sont
			            préservés dans $R$.
		      \end{itemize}
	\end{itemize}
\end{theoreme}

\begin{remarque}
	Dans la suite, nous utiliserons uniquement des règles qui vérifient ces
	conditions de préservation des G-cartes. %
	Ainsi, lorsque $G$ est une G-carte, sa transformation directe $H$ l'est
	aussi. %
\end{remarque}

D'autres conditions permettent de préserver la cohérence géométrique \cite{bellet2011rule,arnould2022preserving}.

\begin{definition}[Renommage]\label{def:renommage-orbite}%
	Soient $n \in \mathbb{N}$ une dimension topologique, $\omega$ et $\omega'$ deux mots sur $([0,n] \cup \{ \_ \})^*$.

	$\orbit{\omega}$ est dite une \emph{orbite pleine} lorsque $\omega$ est un mot sur $[0,n]^*$, c'est-à-dire sans suppression.

	Une \emph{fonction de renommage} $\orbit{\omega} \mapsto \orbit{\omega'}$, d'une orbite pleine
	$\orbit{\omega_1 \omega_2 ... \omega_k}$, vers $\orbit{\omega'_1 \omega'_2 \dots \omega'_k}$ définit
	le renommage des étiquettes topologiques $\omega_1 \mapsto \omega'_1, \omega_2 \mapsto \omega'_2$ \ldots
	$\omega_k \mapsto \omega'_k$. %

	Le \emph{renommage} $(\orbit{\omega} \mapsto \orbit{\omega'})(G\orbit{\omega}(v))$ d'une
	$\orbit{\omega}$-orbite d'un graphe topologique $G=(V,E,s,t,\alpha)$ de dimension $n \in \mathbb{N}$ est le graphe
	$(V,E',s',t',\alpha')$ défini par :
	\begin{itemize}
		\item $E'$ est le sous-ensemble des arcs de $E$ qui n'ont pas été
		      supprimées, c'est-à-dire
		      $E'=\{e \in E \mid (\orbit{\omega} \mapsto \orbit{\omega'})(\alpha(e)) \neq \_ \}$,

		\item $s'$ et $t'$ sont les restrictions respectives de $s$ et $t$ sur
		      $E'$,

		\item $\alpha'$ est définie sur $E'$ pour toute arête $e \in E'$ par
		      $\alpha'(e) = (\orbit{\omega} \mapsto \orbit{\omega'})(\alpha(e))$.
	\end{itemize}
\end{definition}

\begin{definition}[Instanciation d'un motif]\label{def:instanciation-motif}%
	Soient $G$ un graphe topologique de dimension $n \in \mathbb{N}$, $v$ un brin de $G$ et $\orbit{\omega}$ un type d'orbite de dimension $n$.

	L'instanciation d'un motif $M=(V,E,s,t,l,\alpha)$, dont toutes les étiquettes d'orbites ont la même longueur que $\orbit{\omega}$, par l'orbite $G\orbit{\omega}(v)=(V',E',s',t',\alpha')$ est définie par le graphe topologique $M(G\orbit{\omega}(v))=(V'',E'',s'',t'',\alpha'')$~:
	\begin{itemize}
		\item $V^{\prime\prime}= V \times V'$,

		\item $E''=E''_i \cup E''_e$ comprend les \emph{arcs implicites} issus de l'instanciation des nœuds du motif $E''_i = \{ (v, e') \in V \times E' \mid (\orbit{\omega} \mapsto \orbit{\alpha(v)})(\alpha'(e')) \neq \_ \}$ et les \emph{arcs explicites} issus du motif $E''_e = \{ (e, v') \in E \times V' \}$,

		\item tout arc implicite $(v,e') \in E''_i$ a pour source $s''((v,e'))=(v,s'(e'))$, pour cible $t''((v,e'))=(v,t'(e'))$ et pour étiquette $\alpha''((v,e'))=(\orbit{\omega} \mapsto \alpha(v))(\alpha'(e'))$,

		      et tout arc explicite $(e,v') \in E''_e$, a pour source $s''((e,v')) = ((s(e),v')$, pour cible $t''((e,v')) = ((t(e),v')$, et pour étiquette $\alpha''((e, v')) = \alpha(e)$.
	\end{itemize}
\end{definition}

\begin{definition}[Règle transformation de graphe Jerboa]\label{def:regle-transfo-graphe-jerboa}%
	Une \emph{règle} Jerboa  $r: L \longrightarrow R$  est définie par un couple  de motifs avec
	orbites $L$ et $R$ dont toutes  les étiquettes d'orbites ont la même longueur,
	tels    que   $\underline{L} \longrightarrow \underline{R} $ est une règle de transformation de graphe, dont un sous-ensemble des
	nœuds  de $L$  sont  appelés  nœuds d'accroche  (\emph{hooks}  en anglais)  et
	portent des étiquettes d'orbites pleines, c'est-à-dire sans~"$\_$".
\end{definition}

L'application d'une règle Jerboa est l'application de son instanciation, qui est
une règle instanciée. %
Cela nécessite tout d'abord de rechercher dans l'objet à transformer l'orbite
d'instanciation de la règle, d'instancier la règle, de construire le morphisme
de filtrage entre le motif gauche instancié et l'objet, puis finalement
d'appliquer la règle instanciée sur l'objet pour obtenir la transformation
directe. %

\begin{definition}[Orbite d'instanciation]\label{def:orbite-instanciation}%
	Soit $\underline{m} : \underline{L} \rightarrow G$ un morphisme de filtrage
	sur une G-carte $G$, c'est-à-dire un morphisme injectif tel que les nœuds
	d'accroche désignent des orbites isomorphes à renommage près. %
	Plus précisément, pour un nœud d'accroche $h$ de $L$,
	$O = G\orbit{l(h)}(\underline{m}(h))$ est l'orbite d'instanciation. %
	Si la règle $r$ possède un autre nœud d'accroche $h'$, alors
	$O' = G\orbit{l(h')}(\underline{m}(h'))$ est isomorphe à $O$ à renommage près,
	c'est-à-dire que $(l(h') \mapsto l(h))(O')$ est isomorphe à~$O$.
\end{definition}

Par exemple, le morphisme de filtrage qui associe l'unique nœud $n0$ du motif gauche de la règle de triangulation (Figure~\ref{fig:etat-art-model-regle-regle-jerboa-regle-triangulation}) au brin $a$ de la G-carte $G$ (Figure~\ref{fig:etat-art-modelisation-base-regle-instanciation-objet}) définit l'orbite d'instanciation de la Figure~\ref{fig:jerboa-quad-before-triangulation}. %
Le morphisme de filtrage qui associe le même nœud $n0$ au brin $i$ de la G-carte $G$ définit l'orbite d'instanciation de la Figure~\ref{fig:jerboa-tri-before-triangulation}.

\begin{definition}[Filtrage de l'orbite d'instanciation]\label{def:filtrage-orbite-instanciation}%
	Le morphisme de filtrage $m : L(O) \rightarrow G$, s'il existe, est le morphisme qui
	vérifie les propriétés suivantes, pour $L = (V, E, s, t, l, \alpha)$ et
	$O = (V', E', s', t', \alpha')$~:
	\begin{itemize}
		\item pour tout nœud $v \in V$ de $L$,
		      $m_V((v,\underline{m}(h))) = \underline{m}(v)$~;
		\item pour tout arc implicite $(v,e')$, avec $v \in V$ un nœud de $L$ et
		      $e' \in E'$ un arc de l'orbite $O$ non supprimé (c'est-à-dire tel que
		      $\orbit{l(h)} \mapsto \orbit{l(v)} (\alpha'(e')) \neq \_$)~: $m_E((v,e'))$ est
		      l'unique arc de $G$ de source $m_V((v, s'(e')))$ et d'étiquette
		      $\orbit{l(h)} \mapsto \orbit{l(v)}(\alpha'(e'))$ ; et
		      $m_V(t((v, e'))) = t'(m_E((v,e')))$~;
		\item pour tout arc explicite $(e, v')$, avec $e \in E$ un arc de $L$ et
		      $v' \in V'$ un nœud de l'orbite $O$~: $m_E((e, v'))$ est l'unique arc de
		      source $m_V((s(e), v'))$ et d'étiquette $\alpha(e)$~; et
		      $m_V(t(e,v')) = t'(m_E(e,v'))$.
	\end{itemize}
\end{definition}

\begin{remarque}
	Si le morphisme de filtrage $m$ existe, il est unique en raison de la
	contrainte d'arc incident de la G-carte $G$ (définition \ref{def:gcarte}) et
	de la condition d'arc incident du motif $L$ (Théorème~\ref{thm:preservation-coherence-topo-regles}).
\end{remarque}

Comme pour les règles de base, les conditions syntaxiques suivantes sur les règles permettent de garantir la préservation de contraintes des G-cartes lors des transformations directes \cite{belle2012}.

\begin{theoreme}[Préservation de la cohérence topologique~: Règles
		Jerboa]\label{thm:preservation-coherence-topo-regles}%
	Soit une règle $r~: L \longrightarrow R$, $G$ une G-carte et $\underline{m} : \underline{L} \rightarrow G$.

	La transformation directe $G \xlongrightarrow{r, \underline{m}} H$ produit
	une G-carte si les conditions de préservation de la cohérence suivantes sont
	vérifiées :
	\begin{itemize}
		\item \emph{condition de non orientation} : $L$ et $R$ sont non orientés ;
		\item pour toute dimension $i$, un nœud $v$ est source d'un arc implicite
		      si l'étiquette d'orbite de $v$ comprend $i$~;
		\item \emph{condition d'arcs incidents} : pour toute dimension
		      $i \in [0, n]$, la condition d'arcs incidents est celle des règles de
		      base en considérant simultanément les arcs implicites et
		      explicites~:
		      \begin{itemize}
			      \item un nœud préservé de $L \cap R$ est source d'un $i$-arc dans
			            $L$ (implicite ou explicite) si et seulement s'il est
			            source d'un $i$-arc dans $R$ (implicite ou explicite)~;
			      \item tout nœud supprimé de $L \setminus R$ et tout nœud ajouté de
			            $R \setminus L$ est la source d'exactement un $i$-arc (implicite
			            ou explicite) ;
		      \end{itemize}
		\item un nœud $v$ est la source d'un $ijij$-cycle avec $i+2 \geq j$ si :
		      \begin{itemize}
			      \item $v$ est un nœud de $L$ qui contient $i$ et $j$ dans son
			            étiquette d'orbite ;
			      \item $v$ est un nœud de $R$ qui contient $i$ et $j$ dans son
			            étiquette d'orbite et que ces dernières renomment un cycle
			            de $L$, c'est-à-dire tel qu'il existe un nœud $v'$ de $L$
			            avec $l(v) \mapsto l(v') (i) +2 \geq l(v) \mapsto l(v') (j)$ ou
			            $l(v) \mapsto l(v') (j) +2 \geq l(v) \mapsto l(v') (i)$ ;
			      \item $v$ est la source d'un $i$-arc de cible $v'$ (dans $L$ ou
			            $R$) tel que les deux nœuds $v$ et $v'$ ont $j$ dans leurs
			            étiquettes d'orbite non renommé, c'est-à-dire tel que
			            $l(v) \mapsto l(v') (j) = j$ ;
			      \item inversement, $v$ est la source d'un $j$-arc de cible $v'$
			            tel que les deux nœuds $v$ et $v'$ ont $i$ dans leurs
			            étiquettes d'orbites non renommés ;
		      \end{itemize}

		\item \emph{condition de cycle} : pour toutes dimensions $i$ et $j$ tel que
		      $i+2 \geq j$, la condition de cycle est celle des règles de base en considérant les
		      cycles ayant des arcs implicites ou explicites :
		      \begin{itemize}
			      \item un nœud créé $v$ de $R \setminus L$ est la source d'un $ijij$-cycle ;
			      \item si un nœud préservé de $L \cap R$ est la source d'un $ijij$-cycle dans
			            $L$, alors il est la source d'un $ijij$-cycle dans $R$ ;
			      \item si un nœud préservé de $L \cap R$ n'est pas la source d'un $ijij$-cycle
			            dans $L$, alors les $i$-arcs et $j$-arcs dont il est la source dans $L$ sont
			            préservés dans $R$.
		      \end{itemize}
	\end{itemize}
\end{theoreme}
