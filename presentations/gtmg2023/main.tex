  \begin{block}{Brins et arcs}<4>
    Dans une carte généralisée (G-carte), les sommets sont appelés \emph{brins}
    et les arêtes des \emph{arcs}
  \end{block}
\end{frame}

  \begin{columns}
    \begin{column}{.49\textwidth}

        \includegraphics<5>[height=.6\textheight]{house-1-cell-a0-a.pdf}%
      \end{figure}
    \end{column}
    \begin{column}{.49\textwidth}
      \begin{block}{Orbit de type}
        \begin{itemize}
          \item<1-> Face $\orbit{\alpha 0, \alpha 1}$%
          \item<2-> Arête $\orbit{\alpha 0, \alpha 2}$%
          \item<3-> Arête $\orbit{\alpha 0, \alpha 2}$%
          \item<4-> Sommet $\orbit{\alpha 1, \alpha 2}$%
          \item<5-> Arête de face $\orbit{\alpha 0}$%
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \begin{block}{Notation}<5>
    Une orbite, issue d'une G-carte $G$ et de type $\orbit{o}$ incidente à un
    brin $b$ sera notée : $G\orbit{o}(b)$
  \end{block}
\end{frame}

\section{Conclusions et perspectives}
\begin{frame}{Conclusions}
  \begin{itemize}
    \item Déduction d'événement à l'aide de G-cartes et de règles de
          transformation de graphes Jerboa.
    \item Méthode statique de détection agissant avant l'application des règles.
    \item Définition de cas limite.
    \item Hors cas limite, aucun calcul dynamique.
  \end{itemize}
\end{frame}

\begin{frame}{Perspectives}
  À court terme :
  \begin{itemize}
    \item Définir des cas de tests.
    \item Calculer le gain entre le suivi statique et le suivi dynamique.
  \end{itemize}
  \vspace*{1em}
  À plus long terme, intégrer cette méthode pour la génération de journaux de
  bord puis dans le nommage persistant des entités topologique pour le rejeu des
  opérations de modélisation.
\end{frame}
